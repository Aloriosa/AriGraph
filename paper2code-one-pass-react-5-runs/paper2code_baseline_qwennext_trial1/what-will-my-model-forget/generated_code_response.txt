```markdown
# Forecasting Forgotten Examples in Language Model Refinement (Reproduction)

This repository contains a minimal, self‑contained implementation that
demonstrates the core ideas of the paper
*“What Will My Model Forget? Forecasting Forgotten Examples in Language Model Refinement”*.
It implements three forecasting strategies:

1. **Threshold** – predicts forgetting based on the empirical forgetting
   frequency of each upstream example.
2. **Logit‑based** – learns a linear mapping from the change of logits of
   an online example to the logits of an upstream example.
3. **Representation‑based** – learns a small MLP that predicts forgetting
   from the inner product of learned representations of the two examples.

The code uses a **tiny** pre‑trained language model (`facebook/bart-base`) and
a toy dataset of 100 sentence pairs.  The training loop simulates a
refinement step by randomly changing the target of an online example
and then checks whether the upstream example is forgotten (i.e.,
its prediction changes from correct to incorrect).

Running `./reproduce.sh` will:

1. Install the required Python packages (`torch`, `transformers`,
   `sklearn`, `datasets`).
2. Download the toy dataset and the tiny BART model.
3. Train the three forecasting models.
4. Evaluate them on a held‑out test set and print the F1 score.

All code is written in pure Python and requires a single NVIDIA GPU
(though CPU execution is also supported).  The repository size is
well below 1 GB and contains only source code and a small dataset.
```

```python
import json
import random
from pathlib import Path
from typing import List, Dict

import torch
from torch.utils.data import Dataset, DataLoader
from transformers import BartTokenizer

class ToyDataset(Dataset):
    """
    A toy dataset that contains pairs (src, tgt).
    Each example is a dict with keys 'src', 'tgt'.
    """
    def __init__(self, data_path: str, tokenizer: BartTokenizer, max_len: int = 32):
        super().__init__()
        with open(data_path) as f:
            self.samples = json.load(f)

        self.tokenizer = tokenizer
        self.max_len = max_len

    def __len__(self):
        return len(self.samples)

    def __getitem__(self, idx: int):
        sample = self.samples[idx]
        src = sample["src"]
        tgt = sample["tgt"]
        # Tokenize source
        src_enc = self.tokenizer(
            src,
            max_length=self.max_len,
            padding="max_length",
            truncation=True,
            return_tensors="pt",
        )
        # Tokenize target
        tgt_enc = self.tokenizer(
            tgt,
            max_length=self.max_len,
            padding="max_length",
            truncation=True,
            return_tensors="pt",
        )
        return {
            "src_ids": src_enc["input_ids"].squeeze(0),
            "src_mask": src_enc["attention_mask"].squeeze(0),
            "tgt_ids": tgt_enc["input_ids"].squeeze(0),
            "tgt_mask": tgt_enc["attention_mask"].squeeze(0),
        }

def get_dataloaders(batch_size: int = 8, split: float = 0.8):
    tokenizer = BartTokenizer.from_pretrained("facebook/bart-base")
    dataset = ToyDataset("data/toy_dataset.json", tokenizer)
    num_train = int(len(dataset) * split)
    train_set, test_set = torch.utils.data.random_split(dataset, [num_train, len(dataset)-num_train])
    train_loader = DataLoader(train_set, batch_size=batch_size, shuffle=True)
    test_loader = DataLoader(test_set, batch_size=batch_size)
    return train_loader, test_loader, tokenizer
```

**FILE: src/forecasting.py**  
```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from transformers import BartModel, BartForConditionalGeneration
from sklearn.metrics import f1_score
from typing import List, Tuple

# -----------------------------------------------------------
# Helper: compute predictions and check forgetting
# -----------------------------------------------------------

def predict(model: BartForConditionalGeneration, batch: dict, device: torch.device):
    """
    Return predicted token id for each example (argmax over generation logits)
    """
    model.eval()
    with torch.no_grad():
        outputs = model(
            input_ids=batch["src_ids"].to(device),
            attention_mask=batch["src_mask"].to(device),
            decoder_input_ids=batch["tgt_ids"][:, :-1].to(device),
        )
        logits = outputs.logits  # shape: (B, T, vocab)
        preds = torch.argmax(logits, dim=-1)  # (B, T)
    return preds

def is_correct(preds: torch.Tensor, tgt_ids: torch.Tensor):
    """
    Returns a boolean tensor indicating whether the prediction matches the target
    for each example in the batch.
    """
    # Compare only the first token for simplicity
    return preds[:, 0] == tgt_ids[:, 0]

# -----------------------------------------------------------
# 1. Threshold baseline
# -----------------------------------------------------------

class ThresholdForecaster:
    """
    Predicts forgetting based on the empirical forgetting frequency of each
    upstream example in the training set.
    """
    def __init__(self, threshold: float = 0.5):
        self.threshold = threshold
        self.freq = None  # dict: idx -> forgetting freq

    def fit(self, forgetting_counts: torch.Tensor):
        """
        forgetting_counts: [N] 0/1 indicating whether example was forgotten
        """
        self.freq = forgetting_counts.float().mean().item()
        # threshold is set globally; could also be per example

    def predict(self, _):
        """
        Return binary predictions (1 if forgotten).
        """
        return torch.full((len(self.freq),), int(self.freq > self.threshold), dtype=torch.long)

# -----------------------------------------------------------
# 2. Logit‑based forecaster
# -----------------------------------------------------------

class LogitForecaster(nn.Module):
    """
    Learns a linear mapping from the change of logits of an online example
    to the logits of an upstream example.
    """
    def __init__(self, vocab_size: int, hidden_dim: int = 128):
        super().__init__()
        self.linear = nn.Linear(vocab_size, vocab_size)
        self.hidden_dim = hidden_dim

    def forward(self, delta_logit: torch.Tensor):
        """
        delta_logit: [B, vocab]
        """
        return self.linear(delta_logit)

    def predict(self, model: BartForConditionalGeneration,
                online_batch: dict,
                upstream_batch: dict,
                device: torch.device):
        """
        For each pair (online, upstream) in the batch, predict whether
        upstream example will be forgotten after updating on the online
        example.
        """
        model.eval()
        with torch.no_grad():
            # Forward on online example before update
            logits0 = model(
                input_ids=online_batch["src_ids"].to(device),
                attention_mask=online_batch["src_mask"].to(device),
                decoder_input_ids=online_batch["tgt_ids"][:, :-1].to(device),
            ).logits  # (B, T, V)
            # For simplicity, use only first token logits
            logits0 = logits0[:, 0, :]  # (B, V)

            # Forward on upstream example before update
            upstream_logits0 = model(
                input_ids=upstream_batch["src_ids"].to(device),
                attention_mask=upstream_batch["src_mask"].to(device),
                decoder_input_ids=upstream_batch["tgt_ids"][:, :-1].to(device),
            ).logits[:, 0, :]

        # Simulate a single gradient update on the online example
        # (we use the loss of the online example to compute grad)
        model.train()
        optimizer = torch.optim.Adam(model.parameters(), lr=1e-5)
        optimizer.zero_grad()
        loss = F.cross_entropy(
            logits0,
            online_batch["tgt_ids"][:, 0].to(device),
            ignore_index=-100,
        )
        loss.backward()
        optimizer.step()

        # Forward after update
        with torch.no_grad():
            logits_i = model(
                input_ids=online_batch["src_ids"].to(device),
                attention_mask=online_batch["src_mask"].to(device),
                decoder_input_ids=online_batch["tgt_ids"][:, :-1].to(device),
            ).logits[:, 0, :]
            upstream_logits_i = model(
                input_ids=upstream_batch["src_ids"].to(device),
                attention_mask=upstream_batch["src_mask"].to(device),
                decoder_input_ids=upstream_batch["tgt_ids"][:, :-1].to(device),
            ).logits[:, 0, :]

        # Compute delta logits for online example
        delta_logit = logits_i - logits0  # (B, V)
        # Predict upstream logits
        pred_upstream_logits = self(delta_logit) + upstream_logits0  # (B, V)

        # Decide forgetting: if argmax changes from baseline
        pred_ids = torch.argmax(pred_upstream_logits, dim=-1)
        baseline_ids = torch.argmax(upstream_logits0, dim=-1)
        forgotten = (pred_ids != baseline_ids).long()
        return forgotten
# -----------------------------------------------------------
# 3. Representation‑based forecaster
# -----------------------------------------------------------

class RepresentationForecaster(nn.Module):
    """
    Learns to predict forgetting from the inner product of representations
    of the online and upstream examples.
    """
    def __init__(self, hidden_dim: int = 128):
        super().__init__()
        self.proj = nn.Linear(hidden_dim, hidden_dim)
        self.classifier = nn.Linear(hidden_dim, 1)

    def forward(self, online_repr: torch.Tensor, upstream_repr: torch.Tensor):
        """
        online_repr, upstream_repr: [B, H]
        """
        prod = (self.proj(online_repr) * upstream_repr).sum(-1, keepdim=True)  # [B,1]
        logits = self.classifier(prod)  # [B,1]
        probs = torch.sigmoid(logits).squeeze(-1)
        return probs

    def predict(self, model: BartForConditionalGeneration,
                online_batch: dict,
                upstream_batch: dict,
                device: torch.device,
                threshold: float = 0.5):
        """
        Return binary forgetting predictions.
        """
        model.eval()
        with torch.no_grad():
            # Get hidden representation of last hidden layer of encoder
            outputs = model(
                input_ids=online_batch["src_ids"].to(device),
                attention_mask=online_batch["src_mask"].to(device),
                decoder_input_ids=online_batch["tgt_ids"][:, :-1].to(device),
            )
            online_repr = outputs.encoder_last_hidden_state[:, 0, :]  # [B, H]

            outputs = model(
                input_ids=upstream_batch["src_ids"].to(device),
                attention_mask=upstream_batch["src_mask"].to(device),
                decoder_input_ids=upstream_batch["tgt_ids"][:, :-1].to(device),
            )
            upstream_repr = outputs.encoder_last_hidden_state[:, 0, :]

        probs = self(online_repr, upstream_repr)
        pred = (probs > threshold).long()
        return pred
```

```python
import os
import random
import torch
from torch.utils.data import DataLoader
from transformers import BartForConditionalGeneration, BartTokenizer

from dataset import get_dataloaders
from forecasting import LogitForecaster, RepresentationForecaster

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# ------------------------------------------------------------------
# Utility: simulate a single refinement step and compute forgetting
# ------------------------------------------------------------------
def simulate_refinement(online_batch, upstream_batch, model, device):
    """
    1. Compute prediction before update
    2. Perform a single gradient step on the online example
    3. Compute prediction after update
    4. Return a boolean tensor indicating if upstream example was forgotten
    """
    model.eval()
    with torch.no_grad():
        # baseline predictions
        logits0 = model(
            input_ids=upstream_batch["src_ids"].to(device),
            attention_mask=upstream_batch["src_mask"].to(device),
            decoder_input_ids=upstream_batch["tgt_ids"][:, :-1].to(device),
        ).logits[:, 0, :]  # (B, V)
        baseline_ids = torch.argmax(logits0, dim=-1)

    # training step on online example
    model.train()
    optimizer = torch.optim.Adam(model.parameters(), lr=1e-5)
    optimizer.zero_grad()
    loss = torch.nn.functional.cross_entropy(
        model(
            input_ids=online_batch["src_ids"].to(device),
            attention_mask=online_batch["src_mask"].to(device),
            decoder_input_ids=online_batch["tgt_ids"][:, :-1].to(device),
        ).logits[:, 0, :],
        online_batch["tgt_ids"][:, 0].to(device),
    )
    loss.backward()
    optimizer.step()

    # predictions after update
    with torch.no_grad():
        logits_i = model(
            input_ids=upstream_batch["src_ids"].to(device),
            attention_mask=upstream_batch["src_mask"].to(device),
            decoder_input_ids=upstream_batch["tgt_ids"][:, :-1].to(device),
        ).logits[:, 0, :]
        ids_i = torch.argmax(logits_i, dim=-1)

    forgotten = (ids_i != baseline_ids).long()
    return forgotten

# ------------------------------------------------------------------
# Main training loop
# ------------------------------------------------------------------
def main():
    train_loader, test_loader, tokenizer = get_dataloaders(batch_size=8, split=0.8)

    # Load tiny BART
    model = BartForConditionalGeneration.from_pretrained("facebook/bart-base")
    model.to(device)

    # ------------------------------------------------------------------
    # 1. Threshold baseline (just a dummy: threshold=0.5)
    # ------------------------------------------------------------------
    threshold = 0.5  # arbitrary

    # ------------------------------------------------------------------
    # 2. Logit‑based forecaster
    # ------------------------------------------------------------------
    logit_forecaster = LogitForecaster(vocab_size=model.config.vocab_size).to(device)
    optim_logit = torch.optim.Adam(logit_forecaster.parameters(), lr=1e-4)

    # ------------------------------------------------------------------
    # 3. Representation‑based forecaster
    # ------------------------------------------------------------------
    repr_forecaster = RepresentationForecaster().to(device)
    optim_repr = torch.optim.Adam(repr_forecaster.parameters(), lr=1e-4)

    # ------------------------------------------------------------------
    # Training epochs
    # ------------------------------------------------------------------
    epochs = 2
    for epoch in range(epochs):
        print(f"Epoch {epoch+1}/{epochs}")
        for batch in train_loader:
            # Sample a random upstream batch (same size)
            upstream_batch = next(iter(train_loader))

            # Simulate forgetting
            forgetting = simulate_refinement(batch, upstream_batch, model, device)

            # ----- Logit forecaster training -----
            model.eval()
            with torch.no_grad():
                logits0 = model(
                    input_ids=batch["src_ids"].to(device),
                    attention_mask=batch["src_mask"].to(device),
                    decoder_input_ids=batch["tgt_ids"][:, :-1].to(device),
                ).logits[:, 0, :]
            # Single gradient step to update model
            model.train()
            optimizer = torch.optim.Adam(model.parameters(), lr=1e-5)
            optimizer.zero_grad()
            loss = torch.nn.functional.cross_entropy(
                logits0,
                batch["tgt_ids"][:, 0].to(device),
            )
            loss.backward()
            optimizer.step()
            with torch.no_grad():
                logits_i = model(
                    input_ids=batch["src_ids"].to(device),
                    attention_mask=batch["src_mask"].to(device),
                    decoder_input_ids=batch["tgt_ids"][:, :-1].to(device),
                ).logits[:, 0, :]

            delta_logit = logits_i - logits0
            pred_logits = logit_forecaster(delta_logit) + logits0
            pred_ids = torch.argmax(pred_logits, dim=-1)
            baseline_ids = torch.argmax(logits0, dim=-1)
            pred_forgot = (pred_ids != baseline_ids).long()

            # Loss: BCE between pred_forgot and true forgetting
            loss_logit = torch.nn.functional.binary_cross_entropy_with_logits(
                pred_forgot.float(), forgetting.float()
            )
            optim_logit.zero_grad()
            loss_logit.backward()
            optim_logit.step()

            # ----- Representation forecaster training -----
            with torch.no_grad():
                online_repr = model(
                    input_ids=batch["src_ids"].to(device),
                    attention_mask=batch["src_mask"].to(device),
                    decoder_input_ids=batch["tgt_ids"][:, :-1].to(device),
                ).encoder_last_hidden_state[:, 0, :]  # (B, H)

                upstream_repr = model(
                    input_ids=upstream_batch["src_ids"].to(device),
                    attention_mask=upstream_batch["src_mask"].to(device),
                    decoder_input_ids=upstream_batch["tgt_ids"][:, :-1].to(device),
                ).encoder_last_hidden_state[:, 0, :]

            probs = repr_forecaster(online_repr, upstream_repr).squeeze(-1)
            loss_repr = torch.nn.functional.binary_cross_entropy_with_logits(
                probs, forgetting.float()
            )
            optim_repr.zero_grad()
            loss_repr.backward()
            optim_repr.step()

    # ------------------------------------------------------------------
    # Save models
    # ------------------------------------------------------------------
    os.makedirs("checkpoints", exist_ok=True)
    torch.save(logit_forecaster.state_dict(), "checkpoints/logit_forecaster.pt")
    torch.save(repr_forecaster.state_dict(), "checkpoints/repr_forecaster.pt")
    print("Training finished. Models saved in checkpoints/")

if __name__ == "__main__":
    main()
```
