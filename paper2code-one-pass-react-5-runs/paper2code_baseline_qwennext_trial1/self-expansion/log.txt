class FunctionalAdapter(nn.Module):
    def __init__(self, input_dim=768, hidden_dim=128):
        super(FunctionalAdapter, self).__init__()
        self.down_proj = nn.Linear(input_dim, hidden_dim)
        self.up_proj = nn.Linear(hidden_dim, input_dim)
        self.relu = nn.ReLU()
        
    def forward(self, x):
        residual = x
        x = self.down_proj(x)
        x = self.relu(x)
        x = self.up_proj(x)
        return x + residual

class RepresentationDescriptor(nn.Module):
    def __init__(self, input_dim=768, hidden_dim=128):
        super(RepresentationDescriptor, self).__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim)
        )
        self.decoder = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, input_dim)
        )
        
    def forward(self, x):
        encoded = self.encoder(x)
        decoded = self.decoder(encoded)
        return decoded, encoded

class ExpandableWeightingRouter(nn.Module):
    def __init__(self, input_dim=768, max_adapters=10):
        super(ExpandableWeightingRouter, self).__init__()
        self.input_dim = input_dim
        self.max_adapters = max_adapters
        self.adapter_weights = nn.Parameter(torch.randn(input_dim, max_adapters))

        # Compute weights for each adapter
        weights = torch.matmul(x, self.adapter_weights[:, :self.active_adapters])
        weights = torch.softmax(weights, dim=-1)
        return weights
