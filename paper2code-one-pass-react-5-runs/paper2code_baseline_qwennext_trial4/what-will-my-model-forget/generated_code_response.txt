# Forecasting Forgotten Examples – Toy Reproduction

This repository contains a minimal, fully reproducible implementation that
illustrates the core ideas from the paper *“What Will My Model Forget? Forecasting Forgotten Examples in Language Model Refinement”*.
It does **not** aim to reproduce the full experimental results on large
instruction‑tuned models (BART/T5) or the P3/MMLU corpora – those
experiments would require several days of GPU time and many GB of data.
Instead, we provide a lightweight, self‑contained pipeline that

1.  Fine‑tunes a small pretrained model (BART‑Base) on a **single error
    example**.
2.  Measures *forgetting* on a held‑out set of pre‑training examples.
3.  Trains a simple *representation‑based* forecasting model that predicts
    which examples will be forgotten.
4.  Evaluates the forecasting model (F1, precision, recall) and prints the
    results.

All code is written in Python ≥3.10, uses `transformers`, `datasets`,
`torch`, and `scikit‑learn`.  
The script runs on a CPU or GPU; if an NVIDIA GPU is available the
training will automatically use it.

## How to run

```bash
bash reproduce.sh
```

The script will

1.  Install the required packages.
2.  Download the HuggingFace datasets (`wikitext-2` for pre‑training,
    `sst2` for the error set).
3.  Fine‑tune BART‑Base on a single error example.
4.  Compute forgetting on the pre‑training set.
5.  Train the representation‑based forecasting model.
6.  Evaluate and print the metrics.

All intermediate outputs are written to the console and also saved in
`results.txt`.

import random
from pathlib import Path
from typing import List, Tuple

import numpy as np
import torch
from datasets import load_dataset
from transformers import AutoModelForSeq2SeqLM, AutoTokenizer, Trainer, TrainingArguments

# -----------------------------
# Data helpers
# -----------------------------
def load_pretrain_data(
    dataset_name: str = "wikitext",
    dataset_config: str = "wikitext-2-raw-v1",
    split: str = "train",
    max_samples: int = 2000,
) -> List[str]:
    """
    Load a small subset of a text dataset to use as pre‑training examples.
    Returns a list of raw text strings (each string is a sentence).
    """
    ds = load_dataset(dataset_name, dataset_config, split=split)
    # Each row has a 'text' field; split into sentences
    texts = []
    for row in ds:
        for sent in row["text"].split("\n"):
            sent = sent.strip()
            if sent:
                texts.append(sent)
            if len(texts) >= max_samples:
                return texts
    return texts


def load_error_data(
    dataset_name: str = "glue",
    dataset_config: str = "sst2",
    split: str = "validation",
    max_samples: int = 500,
) -> List[Tuple[str, str]]:
    """
    Load a small set of (sentence, label) pairs to use as error examples.
    Labels are converted to strings ("positive"/"negative").
    """
    ds = load_dataset(dataset_name, dataset_config, split=split)
    examples = []
    for row in ds:
        sent = row["sentence"].strip()
        label = "positive" if row["label"] == 1 else "negative"
        examples.append((sent, label))
        if len(examples) >= max_samples:
            break
    return examples


# -----------------------------
# Model utilities
# -----------------------------
def get_tokenizer(model_name: str = "facebook/bart-base"):
    return AutoTokenizer.from_pretrained(model_name)


def get_model(model_name: str = "facebook/bart-base"):
    return AutoModelForSeq2SeqLM.from_pretrained(model_name)


# -----------------------------
# Evaluation helpers
# -----------------------------
def compute_exact_match(
    model: AutoModelForSeq2SeqLM,
    tokenizer: AutoTokenizer,
    inputs: List[str],
    targets: List[str],
    device: torch.device,
    max_length: int = 32,
) -> Tuple[int, int]:
    """
    Return (num_correct, total) for exact match on the given inputs/targets.
    """
    model.eval()
    correct = 0
    total = len(inputs)

    with torch.no_grad():
        for inp, tgt in zip(inputs, targets):
            enc = tokenizer(inp, return_tensors="pt").to(device)
            out = model.generate(
                **enc,
                max_length=max_length,
                num_beams=1,
                early_stopping=True,
            )
            pred = tokenizer.decode(out[0], skip_special_tokens=True).strip()
            if pred.lower() == tgt.lower():
                correct += 1
    return correct, total


def compute_forgetting(
    model: AutoModelForSeq2SeqLM,
    tokenizer: AutoTokenizer,
    pretrain_texts: List[str],
    pretrain_targets: List[str],
    device: torch.device,
) -> List[int]:
    """
    For each pre‑training example, check if it was correct before
    fine‑tuning and after fine‑tuning. Return list of 1 (forgotten) or 0 (not forgotten).
    """
    # Predict before fine‑tune
    before = []
    for txt in pretrain_texts:
        enc = tokenizer(txt, return_tensors="pt").to(device)
        out = model.generate(**enc, max_length=32, num_beams=1, early_stopping=True)
        pred = tokenizer.decode(out[0], skip_special_tokens=True).strip()
        before.append(pred)

    # Fine‑tune on a single error example (provided externally)
    # The fine‑tune step is performed in main.py; here we assume model is already fine‑tuned.

    # Predict after fine‑tune
    after = []
    for txt in pretrain_texts:
        enc = tokenizer(txt, return_tensors="pt").to(device)
        out = model.generate(**enc, max_length=32, num_beams=1, early_stopping=True)
        pred = tokenizer.decode(out[0], skip_special_tokens=True).strip()
        after.append(pred)

    # Compute forgetting (1 if correct before but wrong after)
    forgotten = [int(b != a) for b, a in zip(before, after)]
    return forgotten
```

**FILE: src/forecast.py**
```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.data import DataLoader, Dataset
from tqdm import tqdm
import numpy as np
from sklearn.metrics import f1_score, precision_score, recall_score


class SimpleEncoder(nn.Module):
    """
    Simple MLP encoder that maps a sentence string to a fixed‑dim vector.
    The sentence is tokenized by the provided tokenizer, then averaged
    over token embeddings. The resulting vector is passed through a small
    MLP to produce a d‑dim representation.
    """
    def __init__(self, hidden_dim: int = 128, out_dim: int = 64, vocab_size: int = 30522):
        super().__init__()
        self.embed = nn.Embedding(vocab_size, hidden_dim)
        self.mlp = nn.Sequential(
            nn.Linear(hidden_dim, out_dim),
            nn.ReLU(),
            nn.Linear(out_dim, out_dim),
        )

    def forward(self, input_ids: torch.Tensor) -> torch.Tensor:
        """
        input_ids: [batch, seq_len]
        returns: [batch, out_dim]
        """
        mask = input_ids > 0
        emb = self.embed(input_ids) * mask.unsqueeze(-1)
        summed = emb.sum(dim=1)
        lengths = mask.sum(dim=1).clamp(min=1).unsqueeze(-1)
        avg = summed / lengths
        return self.mlp(avg)


class ForecastDataset(Dataset):
    """
    Dataset that yields pairs (x_i, y_i) and (x_j, y_j) along with
    the ground‑truth forgetting label z_ij.
    """
    def __init__(self, error_examples, pretrain_examples, forgotten_mask):
        """
        error_examples: list of (sentence, target)
        pretrain_examples: list of (sentence, target)
        forgotten_mask: list of 0/1 for each pretrain example
        """
        self.error_examples = error_examples
        self.pretrain_examples = pretrain_examples
        self.forgotten_mask = forgotten_mask

    def __len__(self):
        return len(self.error_examples) * len(self.pretrain_examples)

    def __getitem__(self, idx):
        i = idx // len(self.pretrain_examples)
        j = idx % len(self.pretrain_examples)
        x_i, tgt_i = self.error_examples[i]
        x_j, tgt_j = self.pretrain_examples[j]
        z = self.forgotten_mask[j]
        return x_i, tgt_i, x_j, tgt_j, z


def collate_fn(batch, tokenizer):
    """
    Tokenize batch and return tensors.
    """
    x_i_s, tgt_i_s, x_j_s, tgt_j_s, z = zip(*batch)

    enc_i = tokenizer(list(x_i_s), return_tensors="pt", padding=True, truncation=True)
    enc_j = tokenizer(list(x_j_s), return_tensors="pt", padding=True, truncation=True)
    tgt_i_enc = tokenizer(list(tgt_i_s), return_tensors="pt", padding=True, truncation=True)
    tgt_j_enc = tokenizer(list(tgt_j_s), return_tensors="pt", padding=True, truncation=True)

    return {
        "input_ids_i": enc_i["input_ids"],
        "attention_mask_i": enc_i["attention_mask"],
        "input_ids_j": enc_j["input_ids"],
        "attention_mask_j": enc_j["attention_mask"],
        "labels_i": tgt_i_enc["input_ids"],
        "labels_j": tgt_j_enc["input_ids"],
        "z": torch.tensor(z, dtype=torch.float),
    }


class RepresentationForecaster(nn.Module):
    """
    Black‑box representation‑based forecasting model.
    For each pair (i, j) we compute the dot product of their encoded
    representations and pass through a sigmoid to obtain probability of forgetting.
    """
    def __init__(self, encoder: nn.Module):
        super().__init__()
        self.encoder = encoder

    def forward(self, input_ids_i, attention_mask_i, input_ids_j, attention_mask_j):
        h_i = self.encoder(input_ids_i, attention_mask_i)
        h_j = self.encoder(input_ids_j, attention_mask_j)
        dot = (h_i * h_j).sum(dim=-1)
        prob = torch.sigmoid(dot)
        return prob


def train_rep_forecast(
    forecaster: RepresentationForecaster,
    tokenizer,
    train_loader,
    device,
    epochs: int = 5,
    lr: float = 1e-3,
):
    optimizer = torch.optim.Adam(forecaster.parameters(), lr=lr)
    criterion = nn.BCELoss()

    forecaster.to(device)
    forecaster.train()
    for epoch in range(epochs):
        epoch_loss = 0.0
        for batch in tqdm(train_loader, desc=f"Epoch {epoch+1}/{epochs}"):
            optimizer.zero_grad()
            output = forecaster(
                batch["input_ids_i"].to(device),
                batch["attention_mask_i"].to(device),
                batch["input_ids_j"].to(device),
                batch["attention_mask_j"].to(device),
            )
            loss = criterion(output.squeeze(), batch["z"].to(device))
            loss.backward()
            optimizer.step()
            epoch_loss += loss.item()
        avg_loss = epoch_loss / len(train_loader)
        print(f"  Avg loss: {avg_loss:.4f}")
    return forecaster


def evaluate_forecast(
    forecaster: RepresentationForecaster,
    tokenizer,
    test_loader,
    device,
) -> dict:
    forecaster.eval()
    preds = []
    trues = []
    with torch.no_grad():
        for batch in tqdm(test_loader, desc="Evaluating"):
            prob = forecaster(
                batch["input_ids_i"].to(device),
                batch["attention_mask_i"].to(device),
                batch["input_ids_j"].to(device),
                batch["attention_mask_j"].to(device),
            )
            preds.extend(prob.squeeze().cpu().numpy() > 0.5)
            trues.extend(batch["z"].cpu().numpy())
    f1 = f1_score(trues, preds)
    prec = precision_score(trues, preds)
    rec = recall_score(trues, preds)
    return {"f1": f1, "prec": prec, "rec": rec}

**FILE: src/main.py**
```python
import os
import random
import torch
import numpy as np

from tqdm import tqdm
from transformers import (
    AutoModelForSeq2SeqLM,
    AutoTokenizer,
    Trainer,
    TrainingArguments,
)
from datasets import load_dataset

from utils import (
    load_pretrain_data,
    load_error_data,
    get_tokenizer,
    get_model,
    compute_exact_match,
    compute_forgetting,
)
from forecast import (
    SimpleEncoder,
    RepresentationForecaster,
    ForecastDataset,
    collate_fn,
    train_rep_forecast,
    evaluate_forecast,
)

# -----------------------------
# Reproducibility
# -----------------------------
SEED = 42
random.seed(SEED)
np.random.seed(SEED)
torch.manual_seed(SEED)
torch.cuda.manual_seed_all(SEED)

# -----------------------------
# Settings
# -----------------------------
PRETRAIN_MAX = 2000   # number of pre‑training sentences
ERROR_MAX = 500       # number of error examples
FINE_TUNE_STEPS = 30  # training steps on 1 error example
DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# -----------------------------
# 1. Load data
# -----------------------------
print("=== Loading data ===")
pretrain_texts = load_pretrain_data(max_samples=PRETRAIN_MAX)
pretrain_targets = ["dummy"] * len(pretrain_texts)  # dummy target for seq2seq
error_examples = load_error_data(max_samples=ERROR_MAX)

print(f"Loaded {len(pretrain_texts)} pre‑training sentences")
print(f"Loaded {len(error_examples)} error examples")

# -----------------------------
# 2. Load model & tokenizer
# -----------------------------
print("=== Loading model ===")
tokenizer = get_tokenizer()
model = get_model()
model.to(DEVICE)

# -----------------------------
# 3. Select a single error example to fine‑tune on
# -----------------------------
error_idx = random.randint(0, len(error_examples) - 1)
error_sent, error_label = error_examples[error_idx]
print(f"\nFine‑tuning on error #{error_idx}:")
print(f"  Input sentence: {error_sent[:80]}...")
print(f"  Target label : {error_label}")

# Prepare for fine‑tune: we treat it as a seq2seq task: input -> target
train_dataset = [
    {
        "input_ids": tokenizer(error_sent, return_tensors="pt", truncation=True)["input_ids"][0],
        "labels": tokenizer(error_label, return_tensors="pt", truncation=True)["input_ids"][0],
    }
]
train_loader = torch.utils.data.DataLoader(
    train_dataset,
    batch_size=1,
    shuffle=True,
    collate_fn=lambda x: {
        "input_ids": torch.stack([i["input_ids"] for i in x]),
        "labels": torch.stack([i["labels"] for i in x]),
    },
)

# -----------------------------
# 4. Fine‑tune model on the single example
# -----------------------------
print("\n=== Fine‑tuning ... ===")
optimizer = torch.optim.AdamW(model.parameters(), lr=1e-5)
model.train()
for step in range(FINE_TUNE_STEPS):
    for batch in train_loader:
        optimizer.zero_grad()
        outputs = model(
            input_ids=batch["input_ids"].to(DEVICE),
            labels=batch["labels"].to(DEVICE),
        )
        loss = outputs.loss
        loss.backward()
        optimizer.step()
    if (step + 1) % 10 == 0:
        print(f"  Step {step+1}/{FINE_TUNE_STEPS} - loss: {loss.item():.4f}")

# -----------------------------
# 5. Compute forgetting on pre‑training set
# -----------------------------
print("\n=== Computing forgetting on pre‑training set ===")
forgotten_mask = compute_forgetting(
    model=model,
    tokenizer=tokenizer,
    pretrain_texts=pretrain_texts,
    pretrain_targets=pretrain_targets,
    device=DEVICE,
)

num_forgot = sum(forgotten_mask)
pct_forgot = 100 * num_forgot / len(forgotten_mask)
print(f"  Total examples: {len(forgotten_mask)}")
print(f"  Forgotten: {num_forgot} ({pct_forgot:.2f}%)")

# -----------------------------
# 6. Train representation‑based forecasting model
# -----------------------------
print("\n=== Training representation‑based forecasting model ===")
# Prepare dataset
train_dataset = ForecastDataset(
    error_examples=[error_examples[error_idx]],  # only the single error example
    pretrain_examples=list(zip(pretrain_texts, pretrain_targets)),
    forgotten_mask=forgotten_mask,
)

train_loader = torch.utils.data.DataLoader(
    train_dataset,
    batch_size=64,
    shuffle=True,
    collate_fn=lambda batch: collate_fn(batch, tokenizer),
)

encoder = SimpleEncoder(
    hidden_dim=128,
    out_dim=64,
    vocab_size=tokenizer.vocab_size,
)
forecaster = RepresentationForecaster(encoder)

forecaster = train_rep_forecast(
    forecaster=forecaster,
    tokenizer=tokenizer,
    train_loader=train_loader,
    device=DEVICE,
    epochs=5,
    lr=1e-3,
)

# -----------------------------
# 7. Evaluate forecasting on a hold‑out set of error examples
# -----------------------------
print("\n=== Evaluating forecasting model on remaining error examples ===")
# Use remaining error examples as test set
test_error_examples = error_examples[:error_idx] + error_examples[error_idx+1:]
test_dataset = ForecastDataset(
    error_examples=test_error_examples,
    pretrain_examples=list(zip(pretrain_texts, pretrain_targets)),
    forgotten_mask=forgotten_mask,  # same mask: whether each pretrain example was forgotten
)
test_loader = torch.utils.data.DataLoader(
    test_dataset,
    batch_size=64,
    shuffle=False,
    collate_fn=lambda batch: collate_fn(batch, tokenizer),
)

metrics = evaluate_forecast(
    forecaster=forecaster,
    tokenizer=tokenizer,
    test_loader=test_loader,
    device=DEVICE,
)

print(f"  F1   = {metrics['f1']:.4f}")
print(f"  Prec = {metrics['prec']:.4f}")
print(f"  Rec  = {metrics['rec']:.4f}")

# Persist results
with open("results.txt", "w") as f:
    f.write(f"Forgotten examples: {num_forgot} / {len(forgotten_mask)} ({pct_forgot:.2f}%)\n")
    f.write(f"Forecasting metrics:\n")
    f.write(f"  F1   = {metrics['f1']:.4f}\n")
    f.write(f"  Prec = {metrics['prec']:.4f}\n")
    f.write(f"  Rec  = {metrics['rec']:.4f}\n")

print("\nAll results written to results.txt")
