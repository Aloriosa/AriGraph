# Mechanistic Understanding of Alignment Algorithms (Reproduction)

This repository contains a minimal, fully‑automated reproduction of the key experiments from the paper
> *A Mechanistic Understanding of Alignment Algorithms: A Case Study on DPO and Toxicity*  
> Andrew Lee, Xiaoyan Bai, Itamar Pres, Martin Wattenberg, Jonathan K. Kummerfeld, Rada Mihalcea

The reproduction focuses on the following core components:

1. **Toxicity Probe** – Train a linear probe on the last hidden layer of a pretrained language model
   (GPT‑2 medium) to obtain a “toxicity” direction vector `W_Toxic`.
2. **Toxic Vector Extraction** – Identify the most toxic value vectors in the MLP blocks by cosine
   similarity with `W_Toxic`.
3. **DPO Fine‑Tuning** – Fine‑tune GPT‑2 medium with Direct Preference Optimization (DPO) on a
   pairwise dataset constructed from WikiText‑2 prompts.
4. **Evaluation** – Measure toxicity (via a local BERT toxicity classifier), perplexity on
   WikiText‑2, and a simple F1‑style overlap metric on a small Wikipedia sentence set.

The script `reproduce.sh` installs all dependencies, downloads the required models and datasets,
runs the full pipeline, and outputs a `results.json` file in the repository root.

> **Note**  
> The full paper reports results on GPT‑2 medium *and* Llama‑2‑7B. To keep the reproduction
> lightweight and runnable on a single NVIDIA A10 GPU (24 GB VRAM), we only fine‑tune and
> evaluate GPT‑2 medium. The same pipeline can be applied to Llama‑2‑7B by replacing the model
> name in `main.py`.

```python
#!/usr/bin/env python
import os
import json
import math
import torch
import numpy as np
import pandas as pd
from tqdm import tqdm
from datasets import load_dataset, Dataset
from transformers import (
    AutoTokenizer,
    AutoModelForCausalLM,
    DPOTrainer,
    TrainingArguments,
    DataCollatorForDPO,
    pipeline,
)
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import f1_score

# --------------------------------------------------------------------------- #
# Helper utilities
# --------------------------------------------------------------------------- #
def mean_activation(x):
    """Compute mean activation across sequence length."""
    return x.mean(dim=1)

def cosine_similarity(a, b):
    """Compute cosine similarity between two vectors."""
    a_norm = a / a.norm(dim=-1, keepdim=True)
    b_norm = b / b.norm(dim=-1, keepdim=True)
    return torch.sum(a_norm * b_norm, dim=-1)

def get_last_hidden(model, tokenizer, texts, device):
    """Return mean hidden representation for each text."""
    inputs = tokenizer(texts, return_tensors="pt", padding=True, truncation=True, max_length=512)
    inputs = {k: v.to(device) for k, v in inputs.items()}
    with torch.no_grad():
        outputs = model(**inputs, output_hidden_states=True)
    # Use last hidden state
    last_hidden = outputs.hidden_states[-1]  # (batch, seq_len, hidden)
    return mean_activation(last_hidden)  # (batch, hidden)

def train_toxic_probe(model, tokenizer, dataset, device):
    """Train a simple logistic regression probe on the last hidden layer."""
    print("Training toxicity probe...")
    texts = dataset["text"]
    labels = np.array(dataset["target"])
    hidden = get_last_hidden(model, tokenizer, texts, device).cpu().numpy()
    clf = LogisticRegression(max_iter=200, n_jobs=-1)
    clf.fit(hidden, labels)
    # Return probe vector (coefficients weighted by class)
    w_toxic = torch.tensor(clf.coef_.flatten(), dtype=torch.float32, device=device)
    # Normalize
    w_toxic = w_toxic / w_toxic.norm()
    return w_toxic, clf

def extract_toxic_vectors_gpt2(model, w_toxic, top_k=10):
    """Identify top_k toxic value vectors in GPT‑2 MLP blocks."""
    print("Extracting toxic value vectors (GPT‑2)...")
    toxic_vectors = []
    for i, layer in enumerate(model.model.transformer.h):
        # GPT‑2 MLP: c_fc (key projection), c_proj (value projection)
        key_proj = layer.mlp.c_fc.weight  # (hidden, intermediate)
        val_proj = layer.mlp.c_proj.weight  # (intermediate, hidden)
        # Transpose to get value vectors as columns
        val_vectors = val_proj.t()  # (hidden, intermediate)
        # Compute cosine similarity with w_toxic
        sims = torch.matmul(val_vectors, w_toxic)  # (intermediate)
        top_vals, top_idx = torch.topk(sims, top_k, dim=0)
        for idx, sim in zip(top_idx, top_vals):
            toxic_vectors.append({
                "layer": i,
                "index": idx.item(),
                "similarity": sim.item(),
            })
    return toxic_vectors

def build_pairwise_dataset(prompts, model, tokenizer, w_toxic, device, num_pairs=2000):
    """Construct a small pairwise dataset for DPO."""
    print("Building pairwise dataset for DPO...")
    data = {"prompt": [], "positive": [], "negative": []}
    for text in tqdm(prompts, total=len(prompts)):
        # Positive: greedy continuation
        inputs = tokenizer(text, return_tensors="pt").to(device)
        with torch.no_grad():
            logits = model(**inputs).logits
        next_token = torch.argmax(logits[0, -1]).item()
        pos_cont = tokenizer.decode(next_token, skip_special_tokens=True)
        pos_text = text + pos_cont

        # Negative: sample top‑k tokens, pick the most toxic one
        logits_topk = torch.topk(logits[0, -1], k=10)
        toks = logits_topk.indices.cpu().numpy()
        toks_text = tokenizer.batch_decode(toks, skip_special_tokens=True)
        # Score toxicity via the probe
        toks_hidden = get_last_hidden(model, tokenizer, toks_text, device).cpu()
        scores = (toks_hidden @ w_toxic.unsqueeze(1)).squeeze(1).numpy()
        # Choose token with highest toxicity
        neg_idx = np.argmax(scores)
        neg_cont = toks_text[neg_idx]
        neg_text = text + neg_cont

        data["prompt"].append(text)
        data["positive"].append(pos_text)
        data["negative"].append(neg_text)

        if len(data["prompt"]) >= num_pairs:
            break
    return Dataset.from_dict(data)

def evaluate_toxicity(model, tokenizer, dataset, device, batch_size=8):
    """Compute average toxicity score using a BERT toxicity classifier."""
    print("Evaluating toxicity on dataset...")
    toxic_pipe = pipeline("text-classification", model="unitary/toxic-bert", device=device)
    toxic_scores = []
    for i in tqdm(range(0, len(dataset), batch_size)):
        batch = dataset[i : i + batch_size]["text"]
        preds = toxic_pipe(batch, truncation=True, padding=True)
        # BERT toxicity returns 'LABEL_0' / 'LABEL_1'; we take probability of LABEL_1
        probs = [p[0]["score"] if p[0]["label"] == "LABEL_1" else 0.0 for p in preds]
        toxic_scores.extend(probs)
    return float(np.mean(toxic_scores))

def compute_perplexity(model, tokenizer, dataset, device, batch_size=8):
    """Compute perplexity on the given dataset."""
    print("Computing perplexity...")
    losses = []
    for i in tqdm(range(0, len(dataset), batch_size)):
        batch = dataset[i : i + batch_size]["text"]
        inputs = tokenizer(batch, return_tensors="pt", padding=True, truncation=True, max_length=512).to(device)
        with torch.no_grad():
            outputs = model(**inputs, labels=inputs["input_ids"])
        loss = outputs.loss.item()
        losses.append(loss)
    avg_loss = np.mean(losses)
    perplexity = math.exp(avg_loss)
    return perplexity

def compute_f1_simple(model, tokenizer, dataset, device, batch_size=8):
    """Compute a simple F1‑style overlap between generated and reference."""
    print("Computing simple F1 overlap...")
    overlaps = []
    for i in tqdm(range(0, len(dataset), batch_size)):
        batch = dataset[i : i + batch_size]["text"]
        inputs = tokenizer(batch, return_tensors="pt", padding=True, truncation=True, max_length=512).to(device)
        with torch.no_grad():
            outputs = model.generate(**inputs, max_new_tokens=20, do_sample=False)
        generated = tokenizer.batch_decode(outputs, skip_special_tokens=True)
        # Use reference from dataset (the same text)
        # For illustration, compare token sets
        for ref, gen in zip(batch, generated):
            ref_set = set(ref.split())
            gen_set = set(gen.split())
            tp = len(ref_set & gen_set)
            fp = len(gen_set - ref_set)
            fn = len(ref_set - gen_set)
            precision = tp / (tp + fp) if (tp + fp) > 0 else 0
            recall = tp / (tp + fn) if (tp + fn) > 0 else 0
            f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0
            overlaps.append(f1)
    return float(np.mean(overlaps))

# --------------------------------------------------------------------------- #
# Main reproduction pipeline
# --------------------------------------------------------------------------- #
def main():
    # Device
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print(f"Using device: {device}")

    # Load models
    gpt2_name = "gpt2-medium"
    print(f"Loading GPT‑2 {gpt2_name} ...")
    tokenizer = AutoTokenizer.from_pretrained(gpt2_name)
    model = AutoModelForCausalLM.from_pretrained(gpt2_name).to(device)
    model.eval()

    # 1. Train toxicity probe
    print("Loading Jigsaw toxic comment dataset...")
    jigsaw = load_dataset("jigsaw-toxic-comment-classification-challenge", split="train[:10%]")  # small subset
    w_toxic, probe_clf = train_toxic_probe(model, tokenizer, jigsaw, device)

    # 2. Extract toxic vectors
    toxic_vectors = extract_toxic_vectors_gpt2(model, w_toxic, top_k=5)
    print("Top toxic vectors:")
    for tv in toxic_vectors:
        print(tv)

    # 3. Build pairwise dataset for DPO
    print("Loading WikiText‑2 dataset for prompts...")
    wikitext = load_dataset("wikitext", "wikitext-2-raw-v1", split="train[:1%]")  # small subset
    pairwise_ds = build_pairwise_dataset(
        prompts=wikitext["text"],
        model=model,
        tokenizer=tokenizer,
        w_toxic=w_toxic,
        device=device,
        num_pairs=200
    )

    # 4. Fine‑tune with DPO
    print("Fine‑tuning with DPO...")
    training_args = TrainingArguments(
        output_dir="./dpo_output",
        per_device_train_batch_size=2,
        per_device_eval_batch_size=2,
        num_train_epochs=1,
        logging_steps=10,
        save_steps=200,
        report_to="none",
        fp16=True,
    )
    data_collator = DataCollatorForDPO()
    trainer = DPOTrainer(
        model=model,
        args=training_args,
        data_collator=data_collator,
        train_dataset=pairwise_ds,
        tokenizer=tokenizer,
    )
    trainer.train()

    # 5. Evaluation before and after DPO
    # Load RealToxicityPrompts dataset
    print("Loading RealToxicityPrompts for evaluation...")
    rtp = load_dataset("realtoxicityprompts", split="challenge")
    # Prepare dataset with generated continuations
    def generate_with_model(ds, model, tokenizer, device, prefix=""):
        outputs = []
        for text in tqdm(ds["prompt"], desc=prefix):
            inputs = tokenizer(text, return_tensors="pt").to(device)
            with torch.no_grad():
                out = model.generate(**inputs, max_new_tokens=20, do_sample=False)
            outputs.append(tokenizer.decode(out[0], skip_special_tokens=True))
        return pd.DataFrame({"generated": outputs})

    # Evaluate before DPO
    print("Evaluating pre‑DPO model...")
    pre_dpo_gen = generate_with_model(rtp, model, tokenizer, device, prefix="Pre‑DPO")
    pre_dpo_gen["text"] = rtp["prompt"]
    toxic_before = evaluate_toxicity(pre_dpo_gen, tokenizer, pre_dpo_gen, device)
    ppl_before = compute_perplexity(pre_dpo_gen, tokenizer, pre_dpo_gen, device)
    f1_before = compute_f1_simple(pre_dpo_gen, tokenizer, pre_dpo_gen, device)

    # Load fine‑tuned model
    print("Loading fine‑tuned model for evaluation...")
    finetuned_model = AutoModelForCausalLM.from_pretrained("./dpo_output").to(device)
    finetuned_model.eval()

    # Evaluate after DPO
    print("Evaluating post‑DPO model...")
    post_dpo_gen = generate_with_model(rtp, finetuned_model, tokenizer, device, prefix="Post‑DPO")
    post_dpo_gen["text"] = rtp["prompt"]
    toxic_after = evaluate_toxicity(post_dpo_gen, tokenizer, post_dpo_gen, device)
    ppl_after = compute_perplexity(post_dpo_gen, tokenizer, post_dpo_gen, device)
    f1_after = compute_f1_simple(post_dpo_gen, tokenizer, post_dpo_gen, device)

    # 6. Save results
    results = {
        "toxicity_before_dpo": toxic_before,
        "toxicity_after_dpo": toxic_after,
        "perplexity_before_dpo": ppl_before,
        "perplexity_after_dpo": ppl_after,
        "f1_before_dpo": f1_before,
        "f1_after_dpo": f1_after,
    }
    with open("results.json", "w") as f:
        json.dump(results, f, indent=2)
    print("Results written to results.json")

if __name__ == "__main__":
    main()
```

**Explanation of the Pipeline**

1. **Dependency Installation** – `reproduce.sh` installs all required packages.
2. **Model Loading** – GPT‑2 medium is downloaded from Hugging Face.
3. **Toxic Probe** – A logistic regression classifier is trained on 10 % of the Jigsaw toxic
   dataset. The resulting weight vector `w_toxic` represents the toxicity direction.
4. **Toxic Vector Extraction** – For each MLP block of GPT‑2, the value vectors that are most
   aligned with `w_toxic` are identified.
5. **Pairwise Dataset Construction** – 200 prompt–positive/negative pairs are created from
   WikiText‑2. The negative sample is chosen by picking the token with the highest
   toxicity score according to `w_toxic`.
6. **DPO Fine‑Tuning** – GPT‑2 is fine‑tuned with the DPO trainer for one epoch.
7. **Evaluation** – The pre‑ and post‑DPO models are evaluated on the
   `realtoxicityprompts` dataset for toxicity, perplexity, and a simple F1‑style overlap.
   Results are saved in `results.json`.
