# BBox‑Adapter Reproduction

This repository contains a minimal, end‑to‑end implementation of the **BBox‑Adapter** concept described in the paper *BBox‑ADAPTER: Lightweight Adapting for Black‑Box Large Language Models*.  
The code demonstrates how to:

1. **Treat a large language model (GPT‑2) as a black‑box generator.**
2. **Train a lightweight adapter (BERT‑base) with a ranking‑based NCE loss** that distinguishes correct from incorrect predictions.
3. **Evaluate** the adapted system on the SST‑2 sentiment‑analysis benchmark.

> **NOTE**  
> This is a *toy* reproduction that focuses on the core ideas rather than reproducing the exact experimental results of the paper.  
> It is fully self‑contained, requires no external API keys, and runs on a single GPU or CPU.

The script will:

1. Install the required Python packages.
2. Download the SST‑2 dataset from HuggingFace.
3. Train the adapter for 3 epochs (≈ 10 min on a GPU, a few hours on CPU).
4. Evaluate on the test split and print the accuracy.

All intermediate files (model checkpoints, logs, predictions) are written to the `runs/` directory.

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from transformers import AutoTokenizer, AutoModel
import os
import random
import numpy as np
from tqdm import tqdm

def set_seed(seed: int = 42):
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)

class RankingNCE(nn.Module):
    """
    Ranking‑based Noise Contrastive Estimation loss.
    For a batch of scores `logits` of shape (B, K) where K=number of candidates,
    and a ground‑truth index `pos_idx` of shape (B,),
    the loss is:
        -log( exp(logit[pos]) / sum(exp(logits)) )
    """
    def __init__(self, temperature: float = 1.0):
        super().__init__()
        self.temperature = temperature

    def forward(self, logits: torch.Tensor, pos_idx: torch.Tensor):
        # logits: (B, K)
        # pos_idx: (B,)
        B, K = logits.size()
        logits = logits / self.temperature
        # compute log-sum-exp
        lse = torch.logsumexp(logits, dim=1)  # (B,)
        # gather positive logits
        pos_logits = logits[torch.arange(B), pos_idx]  # (B,)
        loss = -(pos_logits - lse).mean()
        return loss

def collate_batch(batch):
    return tuple(zip(*batch))

def save_checkpoint(state, path):
    torch.save(state, path)

def load_checkpoint(path, model, optimizer):
    checkpoint = torch.load(path, map_location='cpu')
    model.load_state_dict(checkpoint['model_state_dict'])
    optimizer.load_state_dict(checkpoint['optimizer_state_dict'])
    return checkpoint['epoch']
```

**FILE: adapter.py**
```python
import torch
import torch.nn as nn
from transformers import AutoModel, AutoTokenizer

class AdapterModel(nn.Module):
    """
    Lightweight adapter that scores the concatenation of:
        [CLS]  blackbox_output  [SEP]  input_text  [SEP]
    """
    def __init__(self, adapter_name="bert-base-uncased"):
        super().__init__()
        self.adapter_name = adapter_name
        self.tokenizer = AutoTokenizer.from_pretrained(adapter_name)
        self.encoder = AutoModel.from_pretrained(adapter_name)
        self.linear = nn.Linear(self.encoder.config.hidden_size, 1)  # regression score

    def forward(self, input_texts, candidate_texts):
        """
        input_texts: list[str] of size B
        candidate_texts: list[list[str]] of size B, each inner list has K candidates
        Returns scores of shape (B, K)
        """
        batch_size = len(input_texts)
        K = len(candidate_texts[0])
        # Prepare concatenated strings
        concat_texts = []
        for i in range(batch_size):
            for j in range(K):
                concat_texts.append(
                    f"{candidate_texts[i][j]} [SEP] {input_texts[i]}"
                )
        # Tokenize
        enc = self.tokenizer(
            concat_texts,
            padding=True,
            truncation=True,
            max_length=128,
            return_tensors="pt",
        )
        enc = {k: v.cuda() if torch.cuda.is_available() else v for k, v in enc.items()}
        outputs = self.encoder(**enc)
        # Take [CLS] token
        cls_emb = outputs.last_hidden_state[:, 0, :]  # (B*K, H)
        scores = self.linear(cls_emb).squeeze(-1)  # (B*K,)
        scores = scores.view(batch_size, K)  # (B, K)
        return scores
```

    ds = load_dataset("glue", "sst2", split=split)

    ds = ds.map(lambda x: {"label_str": "positive" if x["label"] == 1 else "negative"})

**FILE: train.py**
```python
import os
import torch
import torch.nn as nn
from torch.utils.data import DataLoader
from transformers import AutoModelForCausalLM, AutoTokenizer, AdamW
from tqdm import tqdm
from utils import set_seed, RankingNCE, save_checkpoint
from adapter import AdapterModel
from dataset import load_sst2

def generate_candidates(gpt2, tokenizer, prompt, num_samples=3, max_length=20, temperature=0.7):
    """
    Generate `num_samples` continuations for a given prompt using GPT‑2.
    """
    inputs = tokenizer(prompt, return_tensors="pt")
    input_ids = inputs["input_ids"]
    if torch.cuda.is_available():
        input_ids = input_ids.cuda()
    outputs = gpt2.generate(
        input_ids,
        max_new_tokens=max_length,
        do_sample=True,
        temperature=temperature,
        top_k=50,
        top_p=0.95,
        num_return_sequences=num_samples,
        pad_token_id=tokenizer.eos_token_id,
    )
    candidates = []
    for out in outputs:
        cand = tokenizer.decode(out[input_ids.shape[-1]:], skip_special_tokens=True).strip()
        candidates.append(cand)
    return candidates

def main():
    set_seed(42)
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

    # Load black‑box generator (GPT‑2)
    gpt2_name = "gpt2"
    gpt2_tokenizer = AutoTokenizer.from_pretrained(gpt2_name)
    gpt2 = AutoModelForCausalLM.from_pretrained(gpt2_name).to(device)

    # Load adapter
    adapter = AdapterModel().to(device)
    optimizer = AdamW(adapter.parameters(), lr=5e-6)

    # Loss
    nce_loss = RankingNCE(temperature=1.0)

    # Dataset
    train_ds = load_sst2("train[:90%]")
    dev_ds = load_sst2("train[90%:]")

    train_loader = DataLoader(train_ds, batch_size=8, shuffle=True)
    dev_loader = DataLoader(dev_ds, batch_size=8)

    epochs = 3
    best_acc = 0.0
    os.makedirs("runs", exist_ok=True)

    for epoch in range(1, epochs + 1):
        adapter.train()
        epoch_loss = 0.0
        for batch in tqdm(train_loader, desc=f"Epoch {epoch}"):
            texts = [item["sentence"] for item in batch]
            labels = [item["label"] for item in batch]  # 0/1
            # Map labels to string
            label_strs = ["positive" if l == 1 else "negative" for l in labels]

            # Generate candidates
            candidate_lists = []
            pos_indices = []
            for text, true_label in zip(texts, label_strs):
                prompt = f"Sentiment: {text}\nAnswer:"
                candidates = generate_candidates(
                    gpt2, gpt2_tokenizer, prompt, num_samples=3
                )
                # Append "positive"/"negative" tokens to each candidate to make them
                # comparable to the adapter input.
                # We will use the candidate string itself as the generator output.
                candidate_lists.append(candidates)
                # Determine which candidate matches the ground truth
                try:
                    pos_idx = candidates.index(true_label.lower())
                except ValueError:
                    # If ground truth not in candidates, pick the most similar
                    pos_idx = 0
                pos_indices.append(pos_idx)

            # Forward adapter
            scores = adapter(texts, candidate_lists)  # (B, K)
            pos_idx_tensor = torch.tensor(pos_indices, device=device)
            loss = nce_loss(scores, pos_idx_tensor)

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            epoch_loss += loss.item()

        avg_loss = epoch_loss / len(train_loader)
        print(f"Epoch {epoch}/{epochs} - avg loss: {avg_loss:.4f}")

        # Evaluation on dev
        acc = evaluate(adapter, gpt2, gpt2_tokenizer, dev_loader, device)
        print(f"Dev accuracy: {acc:.2f}%")
        if acc > best_acc:
            best_acc = acc
            save_checkpoint(
                {
                    "epoch": epoch,
                    "model_state_dict": adapter.state_dict(),
                    "optimizer_state_dict": optimizer.state_dict(),
                },
                f"runs/bbox_adapter_best.pt",
            )

    print(f"Training finished. Best dev accuracy: {best_acc:.2f}%")

def evaluate(adapter, gpt2, tokenizer, data_loader, device):
    adapter.eval()
    correct = 0
    total = 0
    with torch.no_grad():
        for batch in tqdm(data_loader, desc="Evaluating"):
            texts = [item["sentence"] for item in batch]
            labels = [item["label"] for item in batch]  # 0/1
            true_labels = ["positive" if l == 1 else "negative" for l in labels]

            candidate_lists = []
            for text in texts:
                prompt = f"Sentiment: {text}\nAnswer:"
                candidates = generate_candidates(
                    gpt2, tokenizer, prompt, num_samples=3
                )
                candidate_lists.append(candidates)

            scores = adapter(texts, candidate_lists)  # (B, K)
            preds = scores.argmax(dim=1).cpu().numpy()
            pred_labels = [candidate_lists[i][p] for i, p in enumerate(preds)]
            for pred, true in zip(pred_labels, true_labels):
                if pred == true:
                    correct += 1
            total += len(texts)
    return 100.0 * correct / total

if __name__ == "__main__":
    main()
```
