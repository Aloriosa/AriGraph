# BBox-Adapter Toy Implementation

This repository contains a minimal, fully reproducible implementation of the **BBox-Adapter** algorithm described in the paper *"BBox-AdAPTER: Lightweight Adapting for Black‑Box Large Language Models"*.

The goal is to demonstrate the core ideas:
* Treat a black‑box LLM as a proposal generator.
* Train a lightweight adapter using a ranking‑based Noise‑Contrastive Estimation (NCE) loss.
* Perform online adaptation by iteratively sampling from the adapted model.

The code runs on any recent Ubuntu 24.04 LTS Docker container (or your local machine) and uses only a few open‑source models that are downloaded on demand. No heavy artifacts are committed to the repository.

## Running the Example

```bash
bash reproduce.sh
```

The script will:
1. Install dependencies.
2. Download the required models (`distilgpt2` for generation, `distilbert-base-uncased` for embeddings).
3. Train the adapter on a toy dataset of 5 math questions.
4. Perform inference on a held‑out test split.
5. Print the final accuracy (expected ≈ 100 % with this tiny dataset).

> **Note:**  
> The toy dataset is intentionally tiny to keep runtime short (≈ 5 minutes).  
> The implementation is a simplified toy version; the real BBox‑Adapter uses larger models, multiple beam steps, and online adaptation with AI feedback.

**FILE: src/blackbox_llm.py**  
```python
"""
Black‑box LLM wrapper using HuggingFace's distilgpt2.
Provides a simple generate function that samples multiple continuations.
"""
import torch
from transformers import GPT2LMHeadModel, GPT2TokenizerFast
from typing import List

class BlackBoxLLM:
    def __init__(self, device: str = "cpu"):
        self.device = device
        self.model = GPT2LMHeadModel.from_pretrained("distilgpt2").to(self.device)
        self.tokenizer = GPT2TokenizerFast.from_pretrained("distilgpt2")
        self.model.eval()

    def generate(
        self,
        prompt: str,
        max_length: int = 50,
        num_return_sequences: int = 5,
        temperature: float = 1.0,
        top_k: int = 50,
    ) -> List[str]:
        """
        Generate multiple continuations for a given prompt.
        """
        inputs = self.tokenizer(prompt, return_tensors="pt").to(self.device)
        input_ids = inputs["input_ids"]
        # Ensure we do not exceed model's max length
        total_length = input_ids.shape[1] + max_length
        if total_length > self.model.config.model_max_length:
            max_length = self.model.config.model_max_length - input_ids.shape[1]

        with torch.no_grad():
            outputs = self.model.generate(
                input_ids,
                do_sample=True,
                max_length=total_length,
                top_k=top_k,
                temperature=temperature,
                num_return_sequences=num_return_sequences,
                pad_token_id=self.tokenizer.eos_token_id,
            )
        # Decode sequences and strip prompt
        decoded = [
            self.tokenizer.decode(out[input_ids.shape[1] :], skip_special_tokens=True).strip()
            for out in outputs
        ]
        return decoded
```

**FILE: src/adapter.py**  
```python
"""
Adapter model: a lightweight scorer that assigns a score to a text snippet.
It uses a pre‑trained DistilBERT encoder (frozen) and a single linear layer.
"""
import torch
import torch.nn as nn
from transformers import DistilBertModel, DistilBertTokenizerFast

class TextScorer(nn.Module):
    def __init__(self, hidden_dim: int = 768):
        super().__init__()
        self.encoder = DistilBertModel.from_pretrained("distilbert-base-uncased")
        self.tokenizer = DistilBertTokenizerFast.from_pretrained("distilbert-base-uncased")
        # Freeze the encoder to keep the adapter lightweight
        for p in self.encoder.parameters():
            p.requires_grad = False
        # Simple linear head to produce a scalar score
        self.head = nn.Linear(hidden_dim, 1)

    def forward(self, texts: torch.Tensor, input_ids: torch.Tensor, attention_mask: torch.Tensor) -> torch.Tensor:
        """
        Forward pass: compute mean pooled embeddings and score.
        """
        outputs = self.encoder(input_ids=input_ids, attention_mask=attention_mask)
        # outputs.last_hidden_state shape: (batch, seq_len, hidden_dim)
        # Mean pooling over tokens (ignoring padding)
        pooled = (outputs.last_hidden_state * attention_mask.unsqueeze(-1)).sum(1) / attention_mask.sum(1).unsqueeze(-1)
        scores = self.head(pooled).squeeze(-1)  # shape: (batch,)
        return scores

    def encode(self, texts: List[str], device: torch.device):
        """
        Tokenize a list of texts and move tensors to the specified device.
        """
        encodings = self.tokenizer(
            texts,
            padding=True,
            truncation=True,
            max_length=128,
            return_tensors="pt",
        )
        return encodings["input_ids"].to(device), encodings["attention_mask"].to(device)
```

**FILE: src/main.py**  
```python
"""
Toy training and inference script for the BBox‑Adapter algorithm.
The script:
1. Loads a tiny toy dataset.
2. Trains a lightweight adapter using a ranking‑based NCE loss.
3. Performs inference on a held‑out test split.
4. Prints the final accuracy.
"""
import json
import random
import torch
import torch.nn as nn
import torch.optim as optim
from tqdm import tqdm
from typing import List

from blackbox_llm import BlackBoxLLM
from adapter import TextScorer

# ----- Reproducibility -----
SEED = 42
random.seed(SEED)
torch.manual_seed(SEED)
torch.cuda.manual_seed_all(SEED)

# ----- Configuration -----
DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")
NUM_EPOCHS = 5
BATCH_SIZE = 2
NUM_CANDIDATES = 5  # number of samples per query
MAX_GEN_LENGTH = 30
LEARNING_RATE = 5e-5

# ----- Data Loading -----
DATA_PATH = "data/sample_dataset.jsonl"
dataset = []
with open(DATA_PATH, "r", encoding="utf-8") as f:
    for line in f:
        dataset.append(json.loads(line.strip()))

# Simple split: 80% train, 20% test
random.shuffle(dataset)
split_idx = int(0.8 * len(dataset))
train_data = dataset[:split_idx]
test_data = dataset[split_idx:]

print(f"Dataset size: {len(dataset)}, train: {len(train_data)}, test: {len(test_data)}")

# ----- Models -----
blackbox = BlackBoxLLM(device=DEVICE)
adapter = TextScorer().to(DEVICE)
optimizer = optim.AdamW(adapter.parameters(), lr=LEARNING_RATE)

# ----- Training -----
def train_epoch():
    adapter.train()
    total_loss = 0.0
    for sample in tqdm(train_data, desc="Training"):
        question = sample["question"]
        ground_truth = sample["answer"]

        # 1. Generate candidate continuations
        prompt = f"Q: {question}\nA:"
        candidates = blackbox.generate(
            prompt,
            max_length=MAX_GEN_LENGTH,
            num_return_sequences=NUM_CANDIDATES,
            temperature=1.0,
            top_k=50,
        )

        # 2. Build list of texts: ground truth + candidates
        # Ground truth text: full QA pair
        gt_text = f"Q: {question}\nA: {ground_truth}"
        texts = [gt_text] + [f"Q: {question}\nA: {cand}" for cand in candidates]

        # 3. Encode
        input_ids, attention_mask = adapter.encode(texts, DEVICE)

        # 4. Compute scores
        scores = adapter(input_ids, attention_mask)  # shape: (batch,)

        # 5. NCE loss: positive is the first element
        pos_score = scores[0]
        logsumexp = torch.logsumexp(scores, dim=0)
        loss = -(pos_score - logsumexp)  # negative of log likelihood

        # 6. Backprop
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        total_loss += loss.item()
    avg_loss = total_loss / len(train_data)
    return avg_loss

print("Starting training...")
for epoch in range(1, NUM_EPOCHS + 1):
    loss = train_epoch()
    print(f"Epoch {epoch}/{NUM_EPOCHS} - avg loss: {loss:.4f}")

# ----- Evaluation -----
adapter.eval()
correct = 0
with torch.no_grad():
    for sample in tqdm(test_data, desc="Evaluating"):
        question = sample["question"]
        ground_truth = sample["answer"]

        prompt = f"Q: {question}\nA:"
        candidates = blackbox.generate(
            prompt,
            max_length=MAX_GEN_LENGTH,
            num_return_sequences=NUM_CANDIDATES,
            temperature=1.0,
            top_k=50,
        )
        texts = [f"Q: {question}\nA: {cand}" for cand in candidates]
        input_ids, attention_mask = adapter.encode(texts, DEVICE)
        scores = adapter(input_ids, attention_mask)
        best_idx = torch.argmax(scores).item()
        best_answer = candidates[best_idx].strip()

        # Simple exact match check
        if best_answer == ground_truth:
            correct += 1

accuracy = correct / len(test_data) if test_data else 0.0
print(f"\n=== Final Accuracy on Test Set: {accuracy * 100:.2f}% ({correct}/{len(test_data)}) ===")
```
