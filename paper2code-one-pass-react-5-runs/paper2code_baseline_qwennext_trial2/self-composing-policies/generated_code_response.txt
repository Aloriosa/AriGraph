I'll create a complete reproduction repository for the "Self-Composing Policies for Scalable Continual Reinforcement Learning" paper.

The paper introduces CompoNet, a novel architecture for continual reinforcement learning that uses self-composing policy modules that grow linearly with the number of tasks.

Based on the paper, CompoNet works by:
1. Starting with a single policy module
2. When a new task arrives, freezing the previous module and adding a new trainable module
3. Each new module has access to the outputs of all previous modules and the current state
4. The new module uses attention mechanisms to selectively combine information from previous policies
5. The architecture grows linearly in parameters with tasks, avoiding quadratic growth

The repository will contain:
1. A Python implementation of CompoNet using PyTorch
2. Implementations of the benchmark environments (Meta-World, SpaceInvaders, Freeway)
3. Training scripts for the experiments
4. A reproduce.sh script that runs the experiments

import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np

class SelfComposingPolicyModule(nn.Module):
    """
    Self-composing policy module for CompoNet.
    Implements the three attention heads and internal policy as described in Section 4.2.
    """
    def __init__(self, state_dim, action_dim, model_dim=256, enc_dim=64):
        super(SelfComposingPolicyModule, self).__init__()
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.model_dim = model_dim
        self.enc_dim = enc_dim
        
        # Output attention head parameters
        self.W_out_Q = nn.Linear(enc_dim, model_dim)
        self.W_out_K = nn.Linear(action_dim, model_dim)
        
        # Input attention head parameters
        self.W_in_Q = nn.Linear(enc_dim, model_dim)
        self.W_in_K = nn.Linear(action_dim, model_dim)
        self.W_in_V = nn.Linear(action_dim, model_dim)
        
        # Internal policy parameters
        self.internal_policy = nn.Sequential(
            nn.Linear(enc_dim + model_dim, model_dim),
            nn.ReLU(),
            nn.Linear(model_dim, model_dim),
            nn.ReLU(),
            nn.Linear(model_dim, action_dim)
        )
        
        # Initialize weights
        self.reset_parameters()
        
    def reset_parameters(self):
        # Initialize weights using Xavier initialization
        nn.init.xavier_uniform_(self.W_out_Q.weight)
        nn.init.xavier_uniform_(self.W_out_K.weight)
        nn.init.xavier_uniform_(self.W_in_Q.weight)
        nn.init.xavier_uniform_(self.W_in_K.weight)
        nn.init.xavier_uniform_(self.W_in_V.weight)
        
        # Initialize biases to zero
        nn.init.zeros_(self.W_out_Q.bias)
        nn.init.zeros_(self.W_out_K.bias)
        nn.init.zeros_(self.W_in_Q.bias)
        nn.init.zeros_(self.W_in_K.bias)
        nn.init.zeros_(self.W_in_V.bias)
        
        # Initialize internal policy weights
        for layer in self.internal_policy:
            if isinstance(layer, nn.Linear):
                nn.init.xavier_uniform_(layer.weight)
            if hasattr(layer, 'bias') and layer.bias is not None:
                nn.init.zeros_(layer.bias)
    
    def forward(self, state, prev_outputs, prev_states=None):
        """
        Forward pass through the module.
        
        Args:
            state: Current state representation (batch_size, state_dim)
            prev_outputs: Outputs from previous modules (batch_size, num_prev_modules, action_dim)
            prev_states: Previous states (optional, batch_size, num_prev_modules, state_dim)
            
        Returns:
            output: Final output of the module (batch_size, action_dim)
        """
        # Encode state to encoder space
        h_s = state  # For now, just use the state as is
        if self.enc_dim != self.state_dim:
            # Project state to encoder dimension
            h_s = F.linear(state, torch.eye(self.enc_dim, self.state_dim))
        
        # Output attention head
        # Query from state
        q_out = self.W_out_Q(h_s)
        
        # Keys from previous outputs + positional encoding
        # For positional encoding, we'll use a simple learned positional encoding
        if prev_outputs is not None:
            # Add positional encoding to previous outputs
            # For simplicity, use a learned positional encoding
            # In practice, this would be a learned matrix of size (num_prev_modules, action_dim)
            # Here we'll use a simple sine encoding
            if prev_outputs.shape[1] > 0:
                # Create positional encoding
                pos_enc = torch.zeros_like(prev_outputs)
                for i in range(prev_outputs.shape[1]):
                    pos_enc[:, i, :] = torch.sin(torch.arange(prev_outputs.shape[1]).float() * 0.01)
                # Add positional encoding
                keys = self.W_out_K(prev_outputs + pos_enc)
            else:
                keys = torch.zeros((prev_outputs.shape[0], 1, self.model_dim))
            
            # Values are the previous outputs
            values = prev_outputs
        else:
            keys = torch.zeros((state.shape[0], 1, self.model_dim))
            values = torch.zeros((state.shape[0], 1, self.action_dim))
        
        # Compute attention weights
        attention_weights = torch.bmm(q_out.unsqueeze(1), keys.transpose(1, 2))
        attention_weights = F.softmax(attention_weights / np.sqrt(self.model_dim), dim=2)
        
        # Apply attention
        output_attention = torch.bmm(attention_weights, values)
        output_attention = output_attention.squeeze(1)
        
        # Input attention head
        q_in = self.W_in_Q(h_s)
        if prev_outputs is not None:
            # Concatenate output from output attention head with previous outputs
            # For input attention head, we need to use the output from output attention head
        else:
            output_attention = torch.zeros((state.shape[0], self.action_dim))
        
        # For input attention head, we'll use the output from output attention head
        # and concatenate with the previous outputs
        if prev_outputs is not None:
            input_attention_inputs = torch.cat([output_attention.unsqueeze(1), prev_outputs], dim=1)
        else:
            input_attention_inputs = torch.cat([output_attention.unsqueeze(1), torch.zeros((state.shape[0], 1, self.action_dim))], dim=1)
        
        # Compute keys and values for input attention head
        keys_in = self.W_in_K(input_attention_inputs)
        values_in = self.W_in_V(input_attention_inputs)
        
        # Compute attention weights
        attention_weights_in = torch.bmm(q_in.unsqueeze(1), keys_in.transpose(1, 2))
        attention_weights_in = F.softmax(attention_weights_in / np.sqrt(self.model_dim), dim=2)
        
        # Apply attention
        input_attention = torch.bmm(attention_weights_in, values_in)
        input_attention = input_attention.squeeze(1)
        
        # Internal policy
        # Concatenate state representation and input attention
        internal_input = torch.cat([h_s, input_attention], dim=1)
        internal_output = self.internal_policy(internal_input)
        
        # Final output: Add internal policy output to output attention output
        final_output = output_attention + internal_output
        
        return final_output

import torch
import torch.nn as nn
import numpy as np
import copy

class CompoNet(nn.Module):
    """
    CompoNet: Self-Composing Policies for Scalable Continual Reinforcement Learning
    Main class implementing the CompoNet architecture.
    """
    def __init__(self, state_dim, action_dim, model_dim=256, enc_dim=64):
        super(CompoNet, self).__init__()
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.model_dim = model_dim
        self.enc_dim = enc_dim
        
        # Initialize with a single module
        self.modules = nn.ModuleList()
        self.add_module()
        
        # For inference, we'll keep track of previous outputs
        self.prev_outputs = None
        self.prev_states = None
        
    def add_module(self):
        """Add a new module to the network."""
        module = SelfComposingPolicyModule(self.state_dim, self.action_dim, self.model_dim, self.enc_dim)
        self.modules.append(module)
        
    def forward(self, state, task_id=None):
        """
        Forward pass through CompoNet.
        
        Args:
            state: Current state representation (batch_size, state_dim)
            task_id: Current task id (optional)
            
        Returns:
            output: Final output of the network (batch_size, action_dim)
        """
        # Get the current module
        current_module = self.modules[-1]
        
        # Get previous outputs from all previous modules
        if self.prev_outputs is None:
            prev_outputs = None
        else:
            # Concatenate all previous outputs
            prev_outputs = torch.stack(self.prev_outputs, dim=1)
        
        # Get the output from the current module
        output = current_module(state, prev_outputs)
        
        # Update previous outputs
        if self.prev_outputs is None:
            self.prev_outputs = [output]
        else:
            self.prev_outputs.append(output)
        
        return output
    
    def reset(self):
        """Reset the network for a new task."""
        self.prev_outputs = None
        self.prev_states = None

import torch
import torch.nn as nn
import numpy as np
from tqdm import tqdm
import os
import pickle

class CompoNetTrainer:
    """
    Trainer for CompoNet.
    """
    def __init__(self, model, optimizer, device='cpu'):
        self.model = model
        self.optimizer = optimizer
        self.device = device
        self.model.to(device)
        
        # Loss function
        self.criterion = nn.MSELoss()
        
        # For tracking metrics
        self.train_losses = []
        self.val_losses = []
        
    def train(self, train_loader, val_loader, epochs=10):
        """Train the model."""
        for epoch in range(epochs):
            print(f"Epoch {epoch + 1}/{epochs}")
            train_loss = self._train_epoch(train_loader)
            val_loss = self._val_epoch(val_loader)
            print(f"Train Loss: {train_loss:.4f}, Val Loss: {val_loss:.4f}")
            self.train_losses.append(train_loss)
            self.val_losses.append(val_loss)
        return self.train_losses, self.val_losses
        
    def _train_epoch(self, data_loader):
        """Train for one epoch."""
        self.model.train()
        total_loss = 0
        for batch_idx, (data, target) in enumerate(data_loader):
            data, target = data.to(self.device), target.to(self.device)
            self.optimizer.zero_grad()
            output = self.model(data)
            loss = self.criterion(output, target)
            loss.backward()
            self.optimizer.step()
            total_loss += loss.item()
        return total_loss / len(data_loader)
        
    def _val_epoch(self, data_loader):
        """Validate for one epoch."""
        self.model.eval()
        total_loss = 0
        with torch.no_grad():
            for batch_idx, (data, target) in enumerate(data_loader):
                data, target = data.to(self.device), target.to(self.device)
            output = self.model(data)
            loss = self.criterion(output, target)
            total_loss += loss.item()
        return total_loss / len(data_loader)
        
    def save_model(self, path):
        """Save the model."""
        torch.save(self.model.state_dict(), path)
        
    def load_model(self, path):
        """Load the model."""
        self.model.load_state_dict(torch.load(path))
        self.model.to(self.device)

import gymnasium as gym
import numpy as np

class CompoNetEnvWrapper:
    """
    Wrapper for CompoNet environments.
    """
    def __init__(self, env_name, max_steps=1000):
        self.env_name = env_name
        self.env = gym.make(env_name)
        self.max_steps = max_steps
        self.current_step = 0
        self.current_task = 0
        self.task_list = []
        self.task_id = 0
        self.task_info = {}
        
    def reset(self, task_id=None):
        """Reset the environment."""
        self.env.reset()
        self.current_step = 0
        self.current_task = task_id if task_id is not None else self.current_task
        self.task_id = self.current_task
        return self._get_obs()
        
    def step(self, action):
        """Step the environment."""
        obs, reward, done, info = self.env.step(action)
        self.current_step += 1
        if self.current_step >= self.max_steps:
            done = True
        return obs, reward, done, info
        
    def _get_obs(self):
        """Get the observation."""
        obs = self.env.observation_space.sample()
        return obs
        
    def get_task_info(self):
        """Get task info."""
        return self.task_info
        
    def get_task_id(self):
        """Get task id."""
        return self.task_id
        
    def get_task_list(self):
        """Get task list."""
        return self.task_list
        
    def set_task_list(self, task_list):
        """Set task list."""
        self.task_list = task_list
        
    def set_task_id(self, task_id):
        """Set task id."""
        self.task_id = task_id
        
    def close(self):
        """Close the environment."""
        self.env.close()

    # Create trainer
    trainer = CompoNetTrainer(model, optimizer, device='cpu')

        torch.utils.data.TensorDataset(

        batch_size=args.batch_size,
        shuffle=False)
