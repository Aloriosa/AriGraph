import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np
import math

class ModularAdapter(nn.Module):
    """
    Modular adapter consisting of a functional adapter and a representation descriptor.
    """
    def __init__(self, input_dim, hidden_dim=64, adapter_type='adapter'):
        super(ModularAdapter, self).__init__()
        self.input_dim = input_dim
        self.hidden_dim = hidden_dim
        self.adapter_type = adapter_type
        
        # Functional adapter (lightweight adapter)
        self.down_proj = nn.Linear(input_dim, hidden_dim)
        self.up_proj = nn.Linear(hidden_dim, input_dim)
        self.dropout = nn.Dropout(0.1)
        self.relu = nn.ReLU()
        
        # Representation descriptor (autoencoder for novelty detection)
        self.encoder = nn.Linear(input_dim, hidden_dim)
        self.decoder = nn.Linear(hidden_dim, input_dim)
        self.ae_dropout = nn.Dropout(0.1)
        
        # Initialize weights
        self._init_weights()
        
    def _init_weights(self):
        """Initialize weights using Xavier initialization"""
        nn.init.xavier_uniform_(self.down_proj.weight)
        nn.init.zeros_(self.down_proj.bias)
        nn.init.xavier_uniform_(self.up_proj.weight)
        nn.init.zeros_(self.up_proj.bias)
        nn.init.xavier_uniform_(self.encoder.weight)
        nn.init.zeros_(self.encoder.bias)
        nn.init.xavier_uniform_(self.decoder.weight)
        nn.init.zeros_(self.decoder.bias)
        
    def forward(self, x, adapter_idx=None, mode='train'):
        """
        Forward pass
        x: input features [batch_size, input_dim]
        adapter_idx: specific adapter index to use (for inference)
        mode: 'train' or 'eval'
        """
        if mode == 'train':
            # Functional adapter output
            adapter_out = self.down_proj(x)
            adapter_out = self.relu(adapter_out)
            adapter_out = self.up_proj(adapter_out)
            adapter_out = self.dropout(adapter_out)
            
            # Representation descriptor (AE)
            ae_hidden = self.encoder(x)
            ae_hidden = self.relu(ae_hidden)
            ae_out = self.decoder(ae_hidden)
            ae_out = self.ae_dropout(ae_out)
            
            # Add residual connection
            output = x + adapter_out
            return output, ae_out, ae_hidden
        else:
            # Inference mode
            adapter_out = self.down_proj(x)
            adapter_out = self.relu(adapter_out)
            adapter_out = self.up_proj(adapter_out)
            adapter_out = self.dropout(adapter_out)
            
            # Representation descriptor
            ae_hidden = self.encoder(x)
            ae_hidden = self.relu(ae_hidden)
            ae_out = self.decoder(ae_hidden)
            ae_out = self.ae_dropout(ae_out)
            
            # Add residual connection
            output = x + adapter_out
            return output, ae_out, ae_hidden

class ExpandableWeightingRouter(nn.Module):
    """
    Expandable weighting router for mixture of adapter outputs
    """
    def __init__(self, input_dim, max_adapters=10):
        super(ExpandableWeightingRouter, self).__init__()
        self.input_dim = input_dim
        self.max_adapters = max_adapters
        self.weight_matrix = nn.Parameter(torch.zeros(input_dim, max_adapters))
        self._init_weights()
        
    def _init_weights(self):
        """Initialize weights"""
        nn.init.xavier_uniform_(self.weight_matrix)
        
    def forward(self, x, adapter_outputs, adapter_mask=None):
        """
        Forward pass
        x: input features [batch_size, input_dim]
        adapter_outputs: list of adapter outputs [batch_size, input_dim]
        adapter_mask: mask for active adapters
        """
        # Compute weights for each adapter
        weights = torch.matmul(x, self.weight_matrix)
        weights = F.softmax(weights, dim=-1)
        
        # Apply weights to adapter outputs
        weighted_outputs = torch.stack(adapter_outputs, dim=-1)
        weighted_output = torch.sum(weights.unsqueeze(-1) * weighted_outputs, dim=-1)
        
        return weighted_output

class SEMA(nn.Module):
    """
    Self-Expansion of pre-trained models with Modularized Adaptation
    """
    def __init__(self, input_dim=768, num_layers=12, hidden_dim=64):
        super(SEMA, self).__init__()
        self.input_dim = input_dim
        self.num_layers = num_layers
        self.hidden_dim = hidden_dim
        
        # Pre-trained ViT (frozen)
        self.vit = nn.Identity()  # Will be replaced with actual ViT in training
        self.vit.eval()  # Freeze ViT
        
        # List of modular adapters for each layer
        self.adapters = nn.ModuleList([ModularAdapter(input_dim, hidden_dim) for _ in range(num_layers)])
        
        # Expandable weighting routers for each layer
        self.routers = nn.ModuleList([ExpandableWeightingRouter(input_dim, max_adapters=10) for _ in range(num_layers)]
        
        # Classification head
        self.classifier = nn.Linear(input_dim, 100)  # CIFAR-100
        self.dropout = nn.Dropout(0.1)
        
        # Representation descriptor statistics for z-score
        self.reconstruction_errors = []  # Track reconstruction errors for each adapter
        self.reconstruction_means = []   # Track means of reconstruction errors
        self.reconstruction_stds = []    # Track stds of reconstruction errors
        
    def add_adapter(self, layer_idx):
        """Add a new adapter at specified layer"""
        # Create new adapter
        new_adapter = ModularAdapter(self.input_dim, self.hidden_dim)
        self.adapters[layer_idx] = new_adapter
        self.adapters.append(new_adapter)
        
        # Expand router
        new_router = ExpandableWeightingRouter(self.input_dim, max_adapters=len(self.adapters))
        self.routers[layer_idx] = new_router
        self.routers.append(new_router)
        
        # Initialize reconstruction error statistics
        self.reconstruction_errors.append([])
        self.reconstruction_means.append(0.0)
        self.reconstruction_stds.append(1.0)
        
    def forward(self, x, mode='train'):
        """
        Forward pass
        x: input image [batch_size, channels, height, width]
        mode: 'train' or 'eval'
        """
        # Pass through ViT (frozen)
        # For simplicity, we'll use a dummy ViT
        # In practice, this would be replaced with actual ViT
        x = x.view(x.size(0), x.size(1), -1)  # Flatten
        x = x.mean(dim=1)  # Global average pooling
        
        adapter_outputs = []
        for i in range(self.num_layers):
            # Get adapter outputs
            adapter_output, ae_output, ae_hidden = self.adapters[i](x, mode=mode)
        return x

import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class ViT(nn.Module):
    """
    Vision Transformer
    """
    def __init__(self, img_size=224, patch_size=16, in_chans=3, embed_dim=768, depth=12, num_heads=12, mlp_ratio=4, num_classes=100):
        super(ViT, self).__init__()
        self.img_size = img_size
        self.patch_size = patch_size
        self.in_chans = in_chans
        self.embed_dim = embed_dim
        self.depth = depth
        self.num_heads = num_heads
        self.mlp_ratio = mlp_ratio
        self.num_classes = num_classes
        
        # Patch embedding
        self.patch_embed = nn.Conv2d(in_chans, embed_dim, kernel_size=patch_size, stride=patch_size)
        
        # Positional embedding
        self.pos_embed = nn.Parameter(torch.zeros(1, (img_size // patch_size) ** 2 + 1, embed_dim))
        
        # Transformer blocks
        self.blocks = nn.ModuleList([
            nn.TransformerEncoderLayer(embed_dim, num_heads, dim_feedforward=int(embed_dim * mlp_ratio), batch_first=True)
            for _ in range(depth)
        ])
        
        # Layer norm
        self.norm = nn.LayerNorm(embed_dim)
        
        # Classification head
        self.head = nn.Linear(embed_dim, num_classes)
        
        # Initialize weights
        self._init_weights()
        
    def _init_weights(self):
        """Initialize weights"""
        # Patch embedding
        nn.init.xavier_uniform_(self.patch_embed.weight)
        nn.init.zeros_(self.patch_embed.bias)
        
        # Positional embedding
        nn.init.zeros_(self.pos_embed)
        
        # Head
        nn.init.zeros_(self.head.weight)
        nn.init.zeros_(self.head.bias)
        
    def forward(self, x):
        """Forward pass"""
        # Patch embedding
        x = self.patch_embed(x)
        
        # Flatten patches
        x = x.flatten(2)
        x = x.transpose(1, 2)
        
        # Add positional embedding
        x = x + self.pos_embed
        
        # Transformer blocks
        for block in self.blocks:
            x = block(x)
        
        # Layer norm
        x = self.norm(x)
        
        # Classification head
        x = x.mean(dim=1)
        x = self.head(x)
        
        return x

def compute_z_score(reconstruction_error, mean, std):
    """
    Compute z-score for reconstruction error
    """
    return (reconstruction_error - mean) / std

def get_reconstruction_error(model, data, device):
    """
    Get reconstruction error using AE
    """
    model.eval()
    with torch.no_grad():
        _, ae_out, _ = model(data)
        error = torch.mean((data - ae_out) ** 2)
    return error.item()

def save_checkpoint(model, optimizer, epoch, path):
    """Save checkpoint"""
    torch.save({
        'epoch': epoch,
        'model_state_dict': model.state_dict(),
        'optimizer_state_dict': optimizer.state_dict(),
    }, path)

    """Load checkpoint"""
    checkpoint = torch.load(path)
    model.load_state_dict(checkpoint['model_state_dict'])
    optimizer.load_state_dict(checkpoint['optimizer_state_dict'])
    epoch = checkpoint['epoch']
    return epoch

            loss += criterion(output, target).item()

# SEMA: Self-Expansion of Pre-trained Models with Mixture of Adapters for Continual Learning

This repository contains the complete reproduction of the paper "Self-Expansion of Pre-trained Models with Mixture of Adapters for Continual Learning" by Wang et al.

## Overview

The paper proposes SEMA (Self-Expansion of pre-trained models with Modularized Adaptation), a novel approach for continual learning that automatically expands pre-trained models with modularized adapters on demand. SEMA uses representation descriptors as distribution shift indicators to trigger expansion signals and an expandable weighting router for composing adapter outputs.

## Key Contributions

1. **Self-Expansion Mechanism**: Automatically decides when and where to expand the model based on distribution shifts detected by representation descriptors.

2. **Modular Adapters**: Each adapter consists of a functional adapter and a representation descriptor that captures the feature distribution.

3. **Expandable Weighting Router**: Learns to create a weighted mixture of adapter outputs.

4. **Sub-linear Expansion Rate**: Achieves sub-linear parameter growth rate compared to linear growth in existing methods.
