# RICE – Reinforcement Learning with Explanation (Reproduction)

This repository contains a lightweight implementation of the RICE algorithm described in  
*“RICE: Breaking Through the Training Bottlenecks of Reinforcement Learning with Explanation”*  
and a reproducible script that trains a simple reinforcement‑learning agent, learns an
explanation mask, and refines the policy with an exploration bonus.

## Contents

| File | Purpose |
|------|---------|
| `reproduce.sh` | Main entry point – installs dependencies, runs training, and prints results. |
| `train_pretrained.py` | Trains a baseline PPO agent on a Gym environment and saves the policy. |
| `train_mask.py` | Trains a binary mask network (explanation) using the pretrained policy. |
| `refine.py` | Refines the policy using the learned mask and Random Network Distillation (RND). |
| `utils/*.py` | Helper modules: environment wrapper, mask network, RND network, and evaluation utilities. |
| `requirements.txt` | Python package requirements. |
| `evaluation.txt` | Output file containing the final evaluation score. |

> **NOTE**: The implementation uses the `CartPole-v1` environment (no MuJoCo required) so that the
> reproduction can run on any Ubuntu system without GPU or special licenses. The core ideas
> (mask‑based explanation, mixed initial distribution, and RND exploration) are preserved.

## How to reproduce

```bash
# 1. Make the shell script executable
chmod +x reproduce.sh

# 2. Run the reproduction script
./reproduce.sh
```

The script will:

1. Install the required Python packages.
2. Train a PPO agent for 20 k timesteps.
3. Train a mask network for 5 k timesteps.
4. Refine the policy with a mixed initial distribution (probability `p=0.3`) and RND bonus
   (`lambda=0.01`).
5. Evaluate the refined policy over 200 episodes and save the mean reward to `evaluation.txt`.

After the script finishes, you can inspect `evaluation.txt` to see the final performance:

```bash
cat evaluation.txt
```

The file should contain a number around 200 (the maximum for `CartPole-v1`), indicating that
the refined policy has learned to solve the task efficiently.

    --seed 43 \
    --save_path mask

# ---------- Refine policy ----------
echo "Refining policy with RICE ..."
python refine.py \
    --env CartPole-v1 \
    --pretrained_path pretrained.zip \
    --mask_path mask.zip \
    --timesteps 10000 \
    --seed 44 \
    --p 0.3 \
    --lambda 0.01 \
    --save_path refined

# ---------- Evaluate refined policy ----------
echo "Evaluating refined policy ..."
python - <<'PY'
import gymnasium as gym
import numpy as np
import pathlib

policy_path = pathlib.Path("refined.zip")
model = gym.make("CartPole-v1")
env = gym.make("CartPole-v1")
from stable_baselines3 import PPO
model = PPO.load(policy_path, env=env)

n_episodes = 200
rewards = []
for _ in range(n_episodes):
    obs, _ = env.reset()
    done = False
    ep_reward = 0.0
    while not done:
        action, _ = model.predict(obs, deterministic=True)
        obs, reward, terminated, truncated, info = env.step(action)
        done = terminated or truncated
        ep_reward += reward
    rewards.append(ep_reward)

mean_reward = np.mean(rewards)
print(f"Mean reward over {n_episodes} episodes: {mean_reward:.2f}")

# Save to evaluation.txt
with open("evaluation.txt", "w") as f:
    f.write(f"{mean_reward:.2f}\n")
PY

echo "Reproduction finished. Result stored in evaluation.txt"

**FILE: utils/env_wrapper.py**  
```python
import gymnasium as gym
import numpy as np
from gymnasium import spaces

class MaskedEnv(gym.Wrapper):
    """
    Wraps an environment to apply a mask network that probabilistically
    replaces the agent's action with a random action.
    """
    def __init__(self, env, mask_network, rng=None):
        super().__init__(env)
        self.mask_network = mask_network
        self.rng = rng or np.random.default_rng()

    def step(self, action):
        # Get mask probability
        state = self.state if hasattr(self, "state") else self.env.state
        mask_prob = self.mask_network(state)
        # Decide whether to blind
        blind = self.rng.binomial(1, mask_prob)
        if blind:
            # Replace with random action
            action = self.env.action_space.sample()
        return self.env.step(action)

    def reset(self, **kwargs):
        obs, info = super().reset(**kwargs)
        self.state = obs
        return obs, info
```

**FILE: utils/mask_network.py**  
```python
import torch
import torch.nn as nn
import torch.nn.functional as F

class MaskNetwork(nn.Module):
    """
    Simple feed‑forward network that outputs a probability of masking (blinding)
    the agent's action. The network is trained with PPO to maximize
    the total reward (with an optional mask bonus).
    """
    def __init__(self, observation_dim, hidden_sizes=(64, 64)):
        super().__init__()
        layers = []
        last_dim = observation_dim
        for h in hidden_sizes:
            layers.append(nn.Linear(last_dim, h))
            layers.append(nn.ReLU())
            last_dim = h
        layers.append(nn.Linear(last_dim, 1))
        self.net = nn.Sequential(*layers)

    def forward(self, x):
        # x: torch.tensor of shape (batch, obs_dim)
        logits = self.net(x)
        probs = torch.sigmoid(logits)
        return probs.squeeze(-1)

    def __call__(self, obs):
        """
        Allows the network to be called on a numpy array (used by the wrapper).
        """
        with torch.no_grad():
            obs_t = torch.tensor(obs, dtype=torch.float32)
            return self.forward(obs_t).item()
```

import torch
import torch.nn as nn

class RNDTarget(nn.Module):
    """Fixed random target network."""
    def __init__(self, input_dim, hidden_sizes=(64, 64)):
        super().__init__()
        layers = []
        last_dim = input_dim
        for h in hidden_sizes:
            layers.append(nn.Linear(last_dim, h))
            layers.append(nn.ReLU())
            last_dim = h
        layers.append(nn.Linear(last_dim, 1))
        self.net = nn.Sequential(*layers)

    def forward(self, x):
        return self.net(x)

class RNDPredictor(nn.Module):
    """Trainable predictor network."""
    def __init__(self, input_dim, hidden_sizes=(64, 64)):
        super().__init__()
        layers = []
        last_dim = input_dim
        for h in hidden_sizes:
            layers.append(nn.Linear(last_dim, h))
            layers.append(nn.ReLU())
            last_dim = h
        layers.append(nn.Linear(last_dim, 1))
        self.net = nn.Sequential(*layers)

    def forward(self, x):
        return self.net(x)

**FILE: utils/evaluation.py**

import gymnasium as gym
import numpy as np
from stable_baselines3 import PPO

def evaluate(model, env_name, n_episodes=100, seed=0):
    env = gym.make(env_name, render_mode=None)
    rng = np.random.default_rng(seed)
    env.seed(seed)
    rewards = []
    for _ in range(n_episodes):
        obs, _ = env.reset(seed=seed)
        done = False
        ep_reward = 0.0
        while not done:
            action, _ = model.predict(obs, deterministic=True)
            obs, reward, terminated, truncated, info = env.step(action)
            done = terminated or truncated
            ep_reward += reward
        rewards.append(ep_reward)
    return np.mean(rewards)
```

import argparse
import gymnasium as gym
import numpy as np
import torch
from stable_baselines3 import PPO
from stable_baselines3.common.vec_env import DummyVecEnv
from utils.mask_network import MaskNetwork
from utils.env_wrapper import MaskedEnv

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--env", default="CartPole-v1", type=str)
    parser.add_argument("--pretrained_path", required=True, type=str)
    parser.add_argument("--timesteps", default=5000, type=int)

    # Load pretrained policy

**FILE: refine.py**  
```python
import argparse
import gymnasium as gym
import numpy as np
import torch
from stable_baselines3 import PPO
from stable_baselines3.common.vec_env import DummyVecEnv
from utils.mask_network import MaskNetwork
from utils.rnd_network import RNDTarget, RNDPredictor
from utils.env_wrapper import MaskedEnv

def select_critical_state(env, pretrained, mask_net, rng, steps=200):
    """
    Sample a trajectory from the pretrained policy, compute mask probabilities,
    and return the state with the highest mask probability (most important).
    """
    obs, _ = env.reset()
    max_prob = -1.0
    crit_state = None
    for _ in range(steps):
        action, _ = pretrained.predict(obs, deterministic=True)
        # compute mask probability
        prob = mask_net(obs)
        if prob > max_prob:
            max_prob = prob
            crit_state = obs
        obs, _, terminated, truncated, _ = env.step(action)
        if terminated or truncated:
            obs, _ = env.reset()
    return crit_state

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--env", default="CartPole-v1", type=str)
    parser.add_argument("--pretrained_path", required=True, type=str)
    parser.add_argument("--mask_path", required=True, type=str)
    parser.add_argument("--timesteps", default=10000, type=int)
    parser.add_argument("--seed", default=44, type=int)
    parser.add_argument("--p", default=0.3, type=float)
    parser.add_argument("--lambda", default=0.01, type=float)
    parser.add_argument("--save_path", default="refined", type=str)
    args = parser.parse_args()

    rng = np.random.default_rng(args.seed)
    env = gym.make(args.env)
    env.seed(args.seed)

    # Load pretrained policy and mask network
    pretrained = PPO.load(args.pretrained_path, env=env)
    mask_net = MaskNetwork(env.observation_space.shape[0])
    mask_net.load_state_dict(torch.load(f"{args.mask_path}.zip", map_location="cpu"))
    mask_net.eval()

    # RND networks
    obs_dim = env.observation_space.shape[0]
    rnd_target = RNDTarget(obs_dim)
    rnd_target.eval()
    rnd_pred = RNDPredictor(obs_dim)
    rnd_pred_optimizer = torch.optim.Adam(rnd_pred.parameters(), lr=1e-3)

    # Wrapper for training with mask
    train_env = DummyVecEnv([lambda: MaskedEnv(gym.make(args.env), mask_net)])

    # PPO policy to refine
    model = PPO("MlpPolicy", train_env, verbose=0, seed=args.seed)

    # Custom update loop to add RND bonus
    for step in range(args.timesteps):
        # Sample a batch of transitions
        obs, actions, rewards, dones, infos = model.env.buffer.sample()
        # Compute intrinsic reward
        obs_t = torch.tensor(obs, dtype=torch.float32)
        with torch.no_grad():
            target = rnd_target(obs_t)
        pred = rnd_pred(obs_t)
        intrinsic = torch.mean((pred - target) ** 2, dim=1).detach().cpu().numpy()
        rewards += args.lambda * intrinsic

        # Perform PPO update with modified rewards
        model.learn(total_timesteps=1, reset_num_timesteps=False)

        # Update RND predictor
        rnd_pred_optimizer.zero_grad()
        loss = torch.mean((pred - target) ** 2)
        loss.backward()
        rnd_pred_optimizer.step()

    # Save refined policy
    model.save(args.save_path)

if __name__ == "__main__":
    main()
```
