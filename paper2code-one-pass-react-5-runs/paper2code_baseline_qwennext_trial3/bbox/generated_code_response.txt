# BBox‑Adapter – Lightweight Adapter for Black‑Box Large Language Models

This repository contains a minimal, reproducible implementation of the **BBox‑Adapter** method described in  
> *BBox‑ADAPTER: Lightweight Adapting for Black‑Box Large Language Models*  
> Haotian Sun et al., ICLR 24.

The goal is to demonstrate how to:

1. **Fine‑tune a small adapter** (≈ 0.1 B parameters) using a ranking‑based Noise Contrastive Estimation (NCE) loss.
2. **Adapt a black‑box LLM** (e.g. OpenAI GPT‑3.5‑Turbo or a local Mixtral‑8×7B) to a downstream task.
3. **Run inference** with a beam‑search that combines the black‑box proposals with the adapter’s score.

> ⚠️  The full experimental setup in the paper uses thousands of examples and an Azure
>  OpenAI fine‑tuning API.  The code below is a *minimal* reproduction that runs on a
>  laptop (or a single A100) and uses a small toy dataset.  It is intended for
>  educational purposes and does **not** reproduce the exact numbers reported in
>  the paper.

**FILE: src/utils.py**  
```python
import json
import os
import time
from typing import List, Dict, Any

import openai
import torch

# ------------------------------------------------------------------
# 1. Dataset utilities
# ------------------------------------------------------------------
def load_jsonl(path: str) -> List[Dict[str, Any]]:
    """Load a JSON list from file."""
    with open(path, "r", encoding="utf-8") as f:
        data = json.load(f)
    return data


# ------------------------------------------------------------------
# 2. OpenAI LLM wrapper
# ------------------------------------------------------------------
def call_llm(model_name: str, prompt: str, max_tokens: int = 200) -> str:
    """
    Call the OpenAI ChatCompletion endpoint.
    Parameters
    ----------
    model_name: str
        e.g. "gpt-3.5-turbo"
    prompt: str
        The full prompt to send.
    max_tokens: int
        Maximum tokens to generate.
    Returns
    -------
    text: str
        The generated text (trimmed).
    """
    # Make sure the key is set
    if "OPENAI_API_KEY" not in os.environ:
        raise RuntimeError("Set OPENAI_API_KEY in the environment")

    response = openai.ChatCompletion.create(
        model=model_name,
        messages=[{"role": "user", "content": prompt}],
        max_tokens=max_tokens,
        temperature=1.0,
        top_p=1.0,
        n=1,
        stop=None,
    )
    return response.choices[0].message.content.strip()


# ------------------------------------------------------------------
# 3. Simple beam search helper
# ------------------------------------------------------------------
def beam_generate(
    prompt: str,
    beam_size: int,
    max_steps: int,
    model_name: str,
    stop_sequences: List[str] = None,
) -> List[str]:
    """
    Generate a list of candidate completions using beam search.
    Each step we call the LLM once per beam and keep the top `beam_size` completions.
    """
    if stop_sequences is None:
        stop_sequences = []

    beams = [prompt]
    scores = [0.0]  # log-probabilities (not used for ranking here)

    for step in range(max_steps):
        new_beams = []
        new_scores = []

        for b, _ in zip(beams, scores):
            # Append a token placeholder; we just ask the model to finish the sentence
            completion = call_llm(model_name, b, max_tokens=20)
            # Simple heuristic: use length as score (longer => better)
            score = len(completion)
            new_beams.append(b + " " + completion)
            new_scores.append(score)

        # Keep top beam_size beams
        sorted_idx = sorted(range(len(new_beams)),
                            key=lambda i: new_scores[i],
                            reverse=True)[:beam_size]
        beams = [new_beams[i] for i in sorted_idx]
        scores = [new_scores[i] for i in sorted_idx]

    # Strip the original prompt and return just the completions
    completions = [b[len(prompt):].strip() for b in beams]
    return completions


# ------------------------------------------------------------------
# 4. Helper to extract answer from LLM output
# ------------------------------------------------------------------
def extract_answer(text: str) -> str:
    """
    Very naive extraction: pick the first number or the last token.
    """
    tokens = text.strip().split()
    if not tokens:
        return ""
    # Find the last token that looks numeric
    for t in reversed(tokens):
        if t.isdigit():
            return t
    # Fallback: last token
    return tokens[-1]
```

**FILE: src/adapter.py**  
```python
import torch
import torch.nn as nn
from transformers import AutoModel, AutoTokenizer

class EnergyAdapter(nn.Module):
    """
    Small encoder that outputs an energy score g_θ(x, y).
    For simplicity we concatenate the input prompt and candidate answer
    and feed them into a DeBERTa encoder followed by a linear layer.
    """
    def __init__(self, encoder_name: str = "microsoft/deberta-v3-base"):
        super().__init__()
        self.tokenizer = AutoTokenizer.from_pretrained(encoder_name)
        self.encoder = AutoModel.from_pretrained(encoder_name)
        hidden = self.encoder.config.hidden_size
        self.scorer = nn.Linear(hidden, 1)

    def forward(self, prompts: torch.Tensor, candidates: torch.Tensor) -> torch.Tensor:
        """
        prompts: Tensor of shape (B, Lp)
        candidates: Tensor of shape (B, Lc)
        Returns: energy scores of shape (B,)
        """
        # Concatenate prompt and candidate
        input_ids = torch.cat([prompts, candidates], dim=-1)
        attention_mask = torch.ones_like(input_ids)
        outputs = self.encoder(input_ids=input_ids,
                               attention_mask=attention_mask,
                               output_hidden_states=False,
                               return_dict=True)
        # Use [CLS] token representation
        cls = outputs.last_hidden_state[:, 0]  # (B, hidden)
        energy = self.scorer(cls).squeeze(-1)  # (B,)
        return energy
```

**FILE: src/trainer.py**  
```python
import argparse
import json
import os
from datetime import datetime
from typing import List, Dict

import torch
import torch.nn.functional as F
from torch.utils.data import DataLoader, Dataset
from tqdm import tqdm

from .adapter import EnergyAdapter
from .utils import load_jsonl, extract_answer

# ------------------------------------------------------------------
# 1. Dataset wrapper
# ------------------------------------------------------------------
class GSM8KDataloader(Dataset):
    def __init__(self, data: List[Dict[str, str]], tokenizer, max_len=128):
        self.data = data
        self.tokenizer = tokenizer
        self.max_len = max_len

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        q = self.data[idx]["question"]
        a = self.data[idx]["answer"]
        # Encode prompt (question)
        prompt_ids = self.tokenizer.encode(q,
                                           add_special_tokens=True,
                                           truncation=True,
                                           max_length=self.max_len,
                                           return_tensors="pt").squeeze(0)
        # Encode answer
        cand_ids = self.tokenizer.encode(a,
                                        add_special_tokens=True,
                                        truncation=True,
                                        max_length=32,
                                        return_tensors="pt").squeeze(0)
        return {"prompt_ids": prompt_ids,
                "cand_ids": cand_ids,
                "answer": a}

# ------------------------------------------------------------------
# 2. Training loop
# ------------------------------------------------------------------
def train(args):
    # Load data
    train_data = load_jsonl(args.train_data)
    dev_data = load_jsonl(args.dev_data)

    # Tokenizer
    from transformers import AutoTokenizer
    tokenizer = AutoTokenizer.from_pretrained(args.model_name)

    train_dataset = GSM8KDataloader(train_data, tokenizer)
    dev_dataset = GSM8KDataloader(dev_data, tokenizer)

    train_loader = DataLoader(train_dataset,
                              batch_size=args.batch_size,
                              shuffle=True,
                              collate_fn=lambda x: collate_fn(x, tokenizer))
    dev_loader = DataLoader(dev_dataset,
                            batch_size=args.batch_size,
                            shuffle=False,
                            collate_fn=lambda x: collate_fn(x, tokenizer))

    # Model
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model = EnergyAdapter(args.model_name).to(device)
    optimizer = torch.optim.AdamW(model.parameters(), lr=args.learning_rate)

    best_acc = 0.0
    os.makedirs(args.output_dir, exist_ok=True)

    for epoch in range(1, args.epochs + 1):
        model.train()
        epoch_loss = 0.0
        for batch in tqdm(train_loader, desc=f"Epoch {epoch}"):
            optimizer.zero_grad()
            prompt_ids = batch["prompt_ids"].to(device)
            cand_ids = batch["cand_ids"].to(device)

            # Positive: ground-truth answer
            pos_score = model(prompt_ids, cand_ids)

            # Negative: generate K random distractors (for toy we just use random tokens)
            k = 4
            neg_ids = torch.randint(0, tokenizer.vocab_size,
                                    (batch_size := pos_score.shape[0], k, 32),
                                    device=device)
            neg_scores = []
            for i in range(k):
                neg_scores.append(model(prompt_ids, neg_ids[:, i, :]))
            neg_scores = torch.stack(neg_scores, dim=1)  # (B, K)

            # NCE loss: log-sum-exp over all samples (pos + neg)
            logits = torch.cat([pos_score.unsqueeze(1), neg_scores], dim=1)  # (B, K+1)
            loss = -F.log_softmax(logits, dim=1)[:, 0].mean()  # negative log prob of pos
            loss.backward()
            optimizer.step()
            epoch_loss += loss.item()

        avg_loss = epoch_loss / len(train_loader)
        print(f"[{datetime.now()}] Epoch {epoch} - loss: {avg_loss:.4f}")

        # Validation
        acc = evaluate(model, dev_loader, tokenizer, device)
        print(f"[{datetime.now()}] Validation accuracy: {acc:.4f}")

        if acc > best_acc:
            best_acc = acc
            torch.save(model.state_dict(), os.path.join(args.output_dir, "adapter.pt"))
            print(f"  -> New best model saved (acc={best_acc:.4f})")

    print(f"Training finished. Best validation accuracy: {best_acc:.4f}")


# ------------------------------------------------------------------
# 3. Collate function
# ------------------------------------------------------------------
def collate_fn(batch, tokenizer):
    """
    Pad sequences to the same length in the batch.
    """
    prompts = [b["prompt_ids"] for b in batch]
    cands = [b["cand_ids"] for b in batch]

    prompt_ids = torch.nn.utils.rnn.pad_sequence(prompts,
                                                 batch_first=True,
                                                 padding_value=tokenizer.pad_token_id)
    cand_ids = torch.nn.utils.rnn.pad_sequence(cands,
                                               batch_first=True,
                                               padding_value=tokenizer.pad_token_id)
    return {"prompt_ids": prompt_ids,
            "cand_ids": cand_ids,
            "answers": [b["answer"] for b in batch]}


# ------------------------------------------------------------------
# 4. Evaluation
# ------------------------------------------------------------------
def evaluate(model, loader, tokenizer, device):
    model.eval()
    correct = 0
    total = 0
    with torch.no_grad():
        for batch in loader:
            prompt_ids = batch["prompt_ids"].to(device)
            cand_ids = batch["cand_ids"].to(device)

            scores = model(prompt_ids, cand_ids)  # (B,)
            preds = scores.argmax(dim=1)  # (B,) 0 means pos

            for pred, gold in zip(preds, batch["answers"]):
                if pred == 0:  # we always use ground-truth as pos
                    correct += 1
                total += 1
    return correct / total


# ------------------------------------------------------------------
# 5. Main
# ------------------------------------------------------------------
if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="BBox‑Adapter Trainer")
    parser.add_argument("--train_data", type=str, required=True)
    parser.add_argument("--dev_data", type=str, required=True)
    parser.add_argument("--model_name", type=str, default="microsoft/deberta-v3-base")
    parser.add_argument("--epochs", type=int, default=3)
    parser.add_argument("--batch_size", type=int, default=8)
    parser.add_argument("--learning_rate", type=float, default=5e-6)
    parser.add_argument("--output_dir", type=str, default="checkpoints/gsm8k")
    parser.add_argument("--log_every", type=int, default=50)
    args = parser.parse_args()
    train(args)
```

**FILE: src/inference.py**  
```python
import argparse
import json
import os
from typing import List

import torch
from tqdm import tqdm

from .adapter import EnergyAdapter
from .utils import load_jsonl, beam_generate, extract_answer

# ------------------------------------------------------------------
def run_inference(args):
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    tokenizer = EnergyAdapter(args.model_name).tokenizer

    # Load adapter
    adapter = EnergyAdapter(args.model_name).to(device)
    adapter.load_state_dict(torch.load(args.adapter_path, map_location=device))
    adapter.eval()

    # Load input data
    data = load_jsonl(args.input_data)

    predictions = []
    correct = 0
    total = 0

    for item in tqdm(data, desc="Inference"):
        question = item["question"]
        gold = item["answer"]

        # Step 1: generate beam candidates from black‑box LLM
        prompt = f"{question}\nAnswer:"
        candidates = beam_generate(prompt,
                                   beam_size=args.beam_size,
                                   max_steps=args.max_steps,
                                   model_name=args.blackbox_model)

        # Encode prompt and candidates
        prompts_ids = tokenizer.encode(question,
                                       add_special_tokens=True,
                                       truncation=True,
                                       max_length=128,
                                       return_tensors="pt").to(device)

        cand_ids = []
        for c in candidates:
            cid = tokenizer.encode(c,
                                   add_special_tokens=True,
                                   truncation=True,
                                   max_length=32,
                                   return_tensors="pt")
            cand_ids.append(cid)
        cand_ids = torch.cat(cand_ids, dim=0).to(device)  # (B, Lc)

        # Compute energies for each candidate
        energies = []
        for i in range(cand_ids.shape[0]):
            energy = adapter(prompts_ids.unsqueeze(0), cand_ids[i:i+1, :]).item()
            energies.append(energy)

        # Pick candidate with highest energy
        best_idx = int(max(range(len(energies)), key=lambda i: energies[i]))
        best_answer = candidates[best_idx]
        pred = extract_answer(best_answer)

        predictions.append({"question": question,
                            "prediction": pred,
                            "gold": gold,
                            "llm_output": best_answer})
        if pred == gold:
            correct += 1
        total += 1

    acc = correct / total
    print(f"Accuracy: {acc:.4f}")

    # Save results
    out = {"model": args.blackbox_model,
           "adapter": args.model_name,
           "beam_size": args.beam_size,
           "max_steps": args.max_steps,
           "accuracy": acc,
           "predictions": predictions}
    os.makedirs(os.path.dirname(args.output_file), exist_ok=True)
    with open(args.output_file, "w", encoding="utf-8") as f:
        json.dump(out, f, indent=2)


# ------------------------------------------------------------------
if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="BBox‑Adapter Inference")
    parser.add_argument("--model_name", type=str, default="microsoft/deberta-v3-base")
    parser.add_argument("--adapter_path", type=str, required=True)
    parser.add_argument("--blackbox_model", type=str, default="gpt-3.5-turbo")
    parser.add_argument("--input_data", type=str, required=True)
    parser.add_argument("--beam_size", type=int, default=3)
    parser.add_argument("--max_steps", type=int, default=3)
    parser.add_argument("--output_file", type=str, default="results/gsm8k_results.json")
    args = parser.parse_args()
    run_inference(args)
```

**NOTE**  
The code above is intentionally simple; it captures the core ideas:

* **Energy model (`g_θ`)** – a tiny transformer encoder that produces a scalar
  energy for a prompt + candidate pair.
* **Ranking‑based NCE loss** – implemented as a log‑softmax over the positive
  (ground‑truth) and a handful of negative candidates.
* **Beam‑search inference** – calls the black‑box LLM to generate multiple
  candidate answers, scores them with the adapter, and selects the best one.
* **OpenAI integration** – uses `openai.ChatCompletion` to get proposals from
  GPT‑3.5‑Turbo.

Feel free to replace the toy dataset, adjust hyper‑parameters, or swap the
black‑box LLM for a local model.  The repository is fully self‑contained and
should run on any machine with Python 3.10+ and a GPU (or CPU).
