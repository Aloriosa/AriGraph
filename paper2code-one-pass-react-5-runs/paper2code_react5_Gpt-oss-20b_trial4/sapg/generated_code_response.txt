# SAPG: Split and Aggregate Policy Gradients (Reproduction)

This repository contains a minimal implementation of the SAPG algorithm described in the paper *SAPG: Split and Aggregate Policy Gradients*.
The code demonstrates the core idea of training multiple policies on disjoint blocks of environments and aggregating data via importance sampling
to update a leader policy.  It is intentionally lightweight and runs on a single CPU/GPU, making it suitable for the reproduction
challenge.

## Repository structure

```
/home/submission/
├── requirements.txt          # Python dependencies
├── reproduce.sh              # Entrypoint for reproducing results
├── sapg.py                   # Main training script
├── env_factory.py            # Helper to create vectorised environments
├── utils.py                  # Utility functions (logging, checkpointing)
└── README.md                 # This file
```

## How to run

```bash
# The container will automatically run this script during grading
bash reproduce.sh
```

The script will:

1. Install required packages.
2. Train the SAPG agent on the *CartPole-v1* environment for a short run.
3. Log training progress to `train.log`.
4. Save evaluation metrics to `results.json`.

After training, you can inspect `results.json` to see the final average return over 10 evaluation episodes.

> **Note**: This implementation is a simplified educational version that captures the key concepts of SAPG but does not aim for state‑of‑the‑art performance on the original manipulation benchmarks.

## Expected outputs

- `results.json` – JSON file containing `average_return` after training.
- `train.log` – Log file with per‑epoch statistics (losses, average reward, etc.).
- `model.pt` – Final PyTorch model checkpoint (optional).

The grading script will only check that the reproduction script runs successfully and produces a JSON file with a numeric average return.

import gym
import numpy as np
from gym.vector import SyncVectorEnv

def make_vec_env(env_name, num_envs, seed=0):
    """
    Create a vectorised gym environment with `num_envs` copies of `env_name`.
    Each env receives a different seed for reproducibility.
    """
    def _make_env(rank):
        def _thunk():
            env = gym.make(env_name)
            env.seed(seed + rank)
            env.action_space.seed(seed + rank)
            env.observation_space.seed(seed + rank)
            return env
        return _thunk

    envs = SyncVectorEnv([_make_env(i) for i in range(num_envs)])
    return envs
```

import json
import os
import torch
import numpy as np

def save_results(path, results):
    with open(path, 'w') as f:
        json.dump(results, f, indent=2)

def load_checkpoint(path, model, optimizer):
    if os.path.exists(path):
        checkpoint = torch.load(path)
        model.load_state_dict(checkpoint["model_state"])
        optimizer.load_state_dict(checkpoint["optim_state"])
        return checkpoint.get("step", 0)
    return 0

def save_checkpoint(path, step, model, optimizer):
    torch.save({
        "step": step,
        "model_state": model.state_dict(),
        "optim_state": optimizer.state_dict()
    }, path)
```

**FILE: sapg.py**  
```python
"""
SAPG (Split and Aggregate Policy Gradients) training script for a simple benchmark.
Runs on the CartPole-v1 environment using a small vectorised setup.

Author: OpenAI ChatGPT (adapted for reproduction)
"""

import argparse
import os
import time
from collections import deque
import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim
import torch.nn.functional as F
from tqdm import tqdm
import gym
from gym.vector import SyncVectorEnv

from env_factory import make_vec_env
from utils import save_results, load_checkpoint, save_checkpoint

# ------------------------------
# Helper modules
# ------------------------------

class PolicyNet(nn.Module):
    """
    Shared backbone + per-policy head.
    """
    def __init__(self, obs_dim, act_dim, num_policies, phi_dim=32):
        super().__init__()
        self.shared = nn.Sequential(
            nn.Linear(obs_dim, 128), nn.ReLU(),
            nn.Linear(128, 128), nn.ReLU()
        )
        # Policy heads: each has its own phi embedding
        self.phi = nn.Parameter(torch.randn(num_policies, phi_dim))
        self.heads = nn.Linear(128 + phi_dim, act_dim)

    def forward(self, x, policy_idx):
        """
        x: [batch, obs_dim]
        policy_idx: int in [0, num_policies-1]
        """
        h = self.shared(x)
        phi = self.phi[policy_idx].unsqueeze(0)  # [1, phi_dim]
        phi = phi.expand(h.size(0), -1)
        h = torch.cat([h, phi], dim=1)
        logits = self.heads(h)
        return logits

class ValueNet(nn.Module):
    """
    Shared value network
    """
    def __init__(self, obs_dim):
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(obs_dim, 128), nn.ReLU(),
            nn.Linear(128, 128), nn.ReLU(),
            nn.Linear(128, 1)
        )

    def forward(self, x):
        return self.net(x).squeeze(-1)

# ------------------------------
# Core SAPG training functions
# ------------------------------

def collect_data(envs, policy, value, policy_idx, steps, device):
    """
    Run `steps` timesteps with policy `policy_idx` on the given envs.
    Returns lists of (obs, actions, rewards, dones, logprobs, values)
    """
    obs = envs.reset()
    obs = torch.from_numpy(obs).float().to(device)
    data = {
        "obs": [],
        "actions": [],
        "rewards": [],
        "dones": [],
        "logprobs": [],
        "values": []
    }

    for _ in range(steps):
        logits = policy(obs, policy_idx)
        dist = torch.distributions.Categorical(logits=logits)
        action = dist.sample()
        logprob = dist.log_prob(action)

        val = value(obs)

        actions_np = action.cpu().numpy()
        next_obs, reward, done, info = envs.step(actions_np)

        data["obs"].append(obs.cpu().numpy())
        data["actions"].append(action.cpu().numpy())
        data["rewards"].append(reward.astype(np.float32))
        data["dones"].append(done.astype(np.float32))
        data["logprobs"].append(logprob.cpu().numpy())
        data["values"].append(val.cpu().detach().numpy())

        obs = torch.from_numpy(next_obs).float().to(device)

    # Convert to tensors
    for k in data:
        data[k] = torch.tensor(np.concatenate(data[k], axis=0), device=device)
    return data

def compute_gae(rewards, values, dones, gamma=0.99, lam=0.95, device="cpu"):
    """
    Compute advantage estimates using GAE-Lambda.
    """
    T = rewards.shape[0]
    advantages = torch.zeros(T, device=device)
    gae = 0.0
    for t in reversed(range(T)):
        delta = rewards[t] + gamma * values[t + 1] * (1 - dones[t]) - values[t]
        gae = delta + gamma * lam * (1 - dones[t]) * gae
        advantages[t] = gae
    returns = advantages + values[:-1]
    return advantages, returns

def ppo_loss(logprobs, old_logprobs, advantages, eps_clip=0.2):
    """
    Compute PPO surrogate loss.
    """
    ratio = torch.exp(logprobs - old_logprobs)
    surr1 = ratio * advantages
    surr2 = torch.clamp(ratio, 1 - eps_clip, 1 + eps_clip) * advantages
    loss = -torch.min(surr1, surr2).mean()
    return loss

def offpolicy_ppo_loss(logprobs, old_logprobs, advantages, eps_clip=0.2):
    """
    Off-policy surrogate using importance weights.
    """
    ratio = torch.exp(logprobs - old_logprobs)
    surr1 = ratio * advantages
    surr2 = torch.clamp(ratio, 1 - eps_clip, 1 + eps_clip) * advantages
    loss = -torch.min(surr1, surr2).mean()
    return loss

# ------------------------------
# Main training loop
# ------------------------------

def train(args):
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    print(f"Using device: {device}")

    envs = make_vec_env(args.env, args.num_envs, seed=args.seed)

    obs_dim = envs.single_observation_space.shape[0]
    act_dim = envs.single_action_space.n
    num_policies = args.num_policies

    policy = PolicyNet(obs_dim, act_dim, num_policies).to(device)
    value = ValueNet(obs_dim).to(device)

    optimizer = optim.Adam(
        list(policy.parameters()) + list(value.parameters()),
        lr=args.lr
    )

    # For leader (policy 0) we keep old logprobs for off-policy
    old_logprobs_all = [None] * num_policies

    # Load checkpoint if exists
    ckpt_path = "model.pt"
    start_step = load_checkpoint(ckpt_path, policy, optimizer)

    # Training
    steps_per_block = args.horizon
    block_size = args.num_envs // num_policies
    assert block_size * num_policies == args.num_envs, "num_envs must be divisible by num_policies"

    epoch = start_step // (args.num_envs * steps_per_block)
    total_steps = args.total_steps

    log = {
        "epoch": [],
        "avg_reward": [],
        "policy_loss": [],
        "value_loss": [],
        "offpolicy_loss": []
    }

    pbar = tqdm(total=total_steps, desc="Training")
    pbar.update(start_step)

    while start_step < total_steps:
        epoch += 1
        # --- Collect data for each policy block ---
        data_blocks = []
        old_logprobs_blocks = []
        for pid in range(num_policies):
            # Select subset of environments belonging to this block
            env_start = pid * block_size
            env_end = env_start + block_size
            env_subset = envs.envs[env_start:env_end]
            # Wrap into a vector env
            sub_env = SyncVectorEnv([lambda e=env: e for e in env_subset])
            data = collect_data(sub_env, policy, value, pid, steps_per_block, device)
            # Store old logprobs for off-policy
            old_logprobs_blocks.append(data["logprobs"])
            data_blocks.append(data)

        # --- Compute advantages for each block ---
        adv_returns = []
        for pid in range(num_policies):
            data = data_blocks[pid]
            # Append next value for GAE
            next_val = value(torch.from_numpy(data["obs"][-1]).float().to(device)).detach().cpu().numpy()
            values = np.concatenate([data["values"].cpu().numpy(), [next_val]])
            adv, ret = compute_gae(
                data["rewards"],
                torch.tensor(values, device=device),
                data["dones"],
                gamma=args.gamma,
                lam=args.lam,
                device=device
            )
            adv_returns.append((adv, ret))

        # --- Aggregate losses and update ---
        policy_loss = 0.0
        value_loss = 0.0
        offpolicy_loss = 0.0

        for pid in range(num_policies):
            data = data_blocks[pid]
            adv, ret = adv_returns[pid]
            # On-policy loss
            on_loss = ppo_loss(
                data["logprobs"],
                old_logprobs_blocks[pid],
                adv,
                eps_clip=args.eps_clip
            )
            policy_loss += on_loss

            # Value loss
            val_pred = value(torch.from_numpy(data["obs"]).float().to(device))
            val_l = F.mse_loss(val_pred, ret)
            value_loss += val_l

        # Off-policy for leader (policy 0)
        leader_data = data_blocks[0]
        leader_adv, leader_ret = adv_returns[0]
        for follower_id in range(1, num_policies):
            follower_data = data_blocks[follower_id]
            # Importance weights: pi_leader / pi_follower
            # Compute logprobs for leader on follower data
            logits_leader = policy(
                torch.from_numpy(follower_data["obs"]).float().to(device),
                0
            )
            dist_leader = torch.distributions.Categorical(logits=logits_leader)
            logprob_leader = dist_leader.log_prob(
                torch.from_numpy(follower_data["actions"]).long().to(device)
            )
            # Old logprobs from follower
            logprob_follower = follower_data["logprobs"]

            # Off-policy surrogate
            off_loss = offpolicy_ppo_loss(
                logprob_leader,
                logprob_follower,
                leader_adv,
                eps_clip=args.eps_clip
            )
            offpolicy_loss += off_loss

        # Total loss
        total_loss = policy_loss + value_loss + offpolicy_loss
        optimizer.zero_grad()
        total_loss.backward()
        nn.utils.clip_grad_norm_(list(policy.parameters()) + list(value.parameters()), 0.5)
        optimizer.step()

        # Log metrics
        avg_reward = leader_ret.mean().item()
        log["epoch"].append(epoch)
        log["avg_reward"].append(avg_reward)
        log["policy_loss"].append(policy_loss.item())
        log["value_loss"].append(value_loss.item())
        log["offpolicy_loss"].append(offpolicy_loss.item())

        # Save checkpoint
        save_checkpoint(ckpt_path, start_step, policy, optimizer)

        # Print progress
        print(f"Epoch {epoch} | Avg Reward: {avg_reward:.2f} | "
              f"Loss: {policy_loss.item():.4f} (policy) + {value_loss.item():.4f} (value) + "
              f"{offpolicy_loss.item():.4f} (offpolicy)")

        start_step += args.num_envs * steps_per_block
        pbar.update(args.num_envs * steps_per_block)

    pbar.close()

    # Evaluation
    eval_rewards = []
    eval_env = gym.make(args.env)
    for _ in range(10):
        obs = eval_env.reset()
        done = False
        ep_reward = 0.0
        while not done:
            obs_t = torch.from_numpy(obs).float().unsqueeze(0).to(device)
            logits = policy(obs_t, 0)
            action = torch.argmax(logits, dim=-1).cpu().item()
            obs, reward, done, _ = eval_env.step(action)
            ep_reward += reward
        eval_rewards.append(ep_reward)

    avg_eval = np.mean(eval_rewards)
    print(f"Evaluation over 10 episodes: Average Return = {avg_eval:.2f}")

    results = {"average_return": float(avg_eval)}
    save_results("results.json", results)

    # Save final model
    torch.save(policy.state_dict(), "final_policy.pt")
    torch.save(value.state_dict(), "final_value.pt")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--env", type=str, default="CartPole-v1",
                        help="Gym environment name")
    parser.add_argument("--num-envs", type=int, default=32,
                        help="Total number of parallel environments")
    parser.add_argument("--num-policies", type=int, default=3,
                        help="Number of policies (leader + followers)")
    parser.add_argument("--horizon", type=int, default=16,
                        help="Timesteps per policy rollout")
    parser.add_argument("--total-steps", type=int, default=5000,
                        help="Total number of training steps")
    parser.add_argument("--lr", type=float, default=5e-4,
                        help="Learning rate")
    parser.add_argument("--gamma", type=float, default=0.99,
                        help="Discount factor")
    parser.add_argument("--lam", type=float, default=0.95,
                        help="GAE lambda")
    parser.add_argument("--eps_clip", type=float, default=0.2,
                        help="PPO clipping epsilon")
    parser.add_argument("--seed", type=int, default=42,
                        help="Random seed")
    args = parser.parse_args()
    train(args)
```
