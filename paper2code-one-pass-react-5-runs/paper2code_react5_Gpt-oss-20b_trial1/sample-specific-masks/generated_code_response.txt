#!/usr/bin/env bash
set -e

# Update package list and install python3.10
apt-get update && apt-get install -y python3.10 python3.10-venv python3.10-dev

# Create a virtual environment and activate it
python3.10 -m venv venv
source venv/bin/activate

# Upgrade pip
pip install --upgrade pip

# Install required Python packages
pip install --quiet torch torchvision timm tqdm

# Create a directory for logs
mkdir -p logs

# List of supported datasets and backbones
datasets=("CIFAR10" "CIFAR100" "SVHN" "GTSRB" "Flowers102" "DTD" "UCF101" "Food101" "EuroSAT" "OxfordPets" "SUN397")
backbones=("resnet18" "vitb32")

# Run training for each dataset and backbone combination
echo "=== Training started ===" > results.txt
for dataset in "${datasets[@]}"; do
  for backbone in "${backbones[@]}"; do
    echo "Training ${dataset} with ${backbone}..." | tee -a results.txt
    python train.py \
      --mode smm \
      --dataset "$dataset" \
      --backbone "$backbone" \
      --epochs 20 \
      --batch-size 256 \
      --output "logs/${dataset}_${backbone}_smm.txt" >> results.txt 2>&1
    echo "Baseline run..." | tee -a results.txt
    python train.py \
      --mode baseline \
      --dataset "$dataset" \
      --backbone "$backbone" \
      --epochs 20 \
      --batch-size 256 \
      --output "logs/${dataset}_${backbone}_baseline.txt" >> results.txt 2>&1
    echo "-----" >> results.txt
  done
done

echo "=== Training finished ===" >> results.txt
echo "Reproduction finished. Results written to results.txt"

#!/usr/bin/env python3
"""
Train a visual reprogramming model (baseline or SMM) on a specified dataset
and backbone. The script is used by the reproduction shell script.
"""

import argparse
import os
import random
import numpy as np
import torch
import torch.nn as nn
import torch.backends.cudnn as cudnn
import torch.optim as optim
import torchvision
import torchvision.transforms as transforms
from tqdm import tqdm

# --------------------------------------------------------------------------- #
# Utilities
# --------------------------------------------------------------------------- #
def set_random_seed(seed=42):
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)
    cudnn.deterministic = True
    cudnn.benchmark = False

def get_device():
    return torch.device("cuda" if torch.cuda.is_available() else "cpu")

# --------------------------------------------------------------------------- #
# Model components
# --------------------------------------------------------------------------- #
class MaskGenerator(nn.Module):
    """
    Lightweight 5‑layer CNN that outputs a 3‑channel mask.
    The spatial size is reduced by 2^l (default l=2).
    """
    def __init__(self, in_channels=3, l=2):
        super().__init__()
        self.l = l
        self.conv1 = nn.Conv2d(in_channels, 32, kernel_size=3, padding=1)
        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
        self.conv3 = nn.Conv2d(64, 128, kernel_size=3, padding=1)
        self.conv4 = nn.Conv2d(128, 64, kernel_size=3, padding=1)
        self.conv5 = nn.Conv2d(64, 3, kernel_size=3, padding=1)
        self.pool = nn.MaxPool2d(kernel_size=2)
        self.relu = nn.ReLU(inplace=True)

    def forward(self, x):
        # x: (B, 3, H, W)
        x = self.relu(self.conv1(x))
        x = self.pool(x)  # /2
        x = self.relu(self.conv2(x))
        x = self.pool(x)  # /4
        x = self.relu(self.conv3(x))
        x = self.pool(x)  # /8
        x = self.relu(self.conv4(x))
        x = self.relu(self.conv5(x))
        return x  # (B, 3, H//8, W//8)

# --------------------------------------------------------------------------- #
# VR Model
# --------------------------------------------------------------------------- #
class VRModel(nn.Module):
    """
    Wrapper that applies visual reprogramming to a frozen backbone.
    """
    def __init__(self, backbone, mode="baseline", l=2, device="cpu"):
        super().__init__()
        self.device = device
        self.backbone = backbone
        for p in self.backbone.parameters():
            p.requires_grad = False
        self.backbone.eval()

        self.mode = mode
        if mode == "smm":
            self.mask_gen = MaskGenerator(l=l).to(device)
            self.delta = nn.Parameter(torch.zeros(1, 3, *self.input_size, device=device))
        else:  # baseline
            self.delta = nn.Parameter(torch.zeros(1, 3, *self.input_size, device=device))
            # shared mask: all ones (full watermark)
            self.register_buffer("shared_mask", torch.ones(1, 3, *self.input_size, device=device))

        # For patch‑wise interpolation
        self.l = l

    def forward(self, x):
        """
        x: (B, 3, H, W) where H=W= input_size
        """
        B = x.size(0)
        if self.mode == "baseline":
            mask = self.shared_mask
        else:  # smm
            small_mask = self.mask_gen(x)  # (B, 3, H//8, W//8)
            mask = nn.functional.interpolate(small_mask, size=self.input_size, mode="nearest")
        # Reprogrammed input
        x_reprog = x + self.delta * mask
        # Forward through backbone
        with torch.no_grad():
            logits = self.backbone(x_reprog)
        return logits

# --------------------------------------------------------------------------- #
# Data handling
# --------------------------------------------------------------------------- #
def get_dataloaders(dataset_name, input_size, batch_size):
    """
    Returns train and test dataloaders for the specified dataset.
    """
    if dataset_name == "CIFAR10":
        num_classes = 10
    elif dataset_name == "CIFAR100":
        num_classes = 100
    else:
        # For other datasets torchvision uses the original number of classes
        num_classes = getattr(torchvision.datasets, dataset_name).num_classes

    normalize = transforms.Normalize(mean=[0.485, 0.456, 0.406],
                                     std =[0.229, 0.224, 0.225])

    train_transform = transforms.Compose([
        transforms.Resize(input_size),
        transforms.RandomCrop(input_size, padding=4),
        transforms.RandomHorizontalFlip(),
        transforms.ToTensor(),
        normalize,
    ])

    test_transform = transforms.Compose([
        transforms.Resize(input_size),
        transforms.CenterCrop(input_size),
        transforms.ToTensor(),
        normalize,
    ])

    # Dynamically import dataset class
    DatasetCls = getattr(torchvision.datasets, dataset_name)
    train_set = DatasetCls(root='./data', train=True, download=True, transform=train_transform)
    test_set  = DatasetCls(root='./data', train=False, download=True, transform=test_transform)

    train_loader = torch.utils.data.DataLoader(train_set, batch_size=batch_size,
                                               shuffle=True, num_workers=4, pin_memory=True)
    test_loader  = torch.utils.data.DataLoader(test_set, batch_size=batch_size,
                                               shuffle=False, num_workers=4, pin_memory=True)
    return train_loader, test_loader, num_classes

# --------------------------------------------------------------------------- #
# Training utilities
# --------------------------------------------------------------------------- #
def random_label_mapping(num_target_classes, num_pretrained_classes=1000):
    """
    Create a random one‑to‑one mapping from target classes to a subset of pretrained classes.
    """
    indices = torch.randperm(num_pretrained_classes)[:num_target_classes]
    mapping = {i: int(indices[i]) for i in range(num_target_classes)}
    return mapping

def map_labels(target_labels, mapping):
    """
    Convert target labels to pretrained labels using the mapping.
    """
    return torch.tensor([mapping[int(l)] for l in target_labels.numpy()], device=target_labels.device)

def train_one_epoch(model, loader, criterion, optimizer, device):
    model.train()
    running_loss = 0.0
    for inputs, labels in tqdm(loader, leave=False):
        inputs = inputs.to(device)
        labels = labels.to(device)

        optimizer.zero_grad()
        logits = model(inputs)
        # Map labels
        mapped_labels = map_labels(labels, model.mapping)
        loss = criterion(logits, mapped_labels)
        loss.backward()
        optimizer.step()

        running_loss += loss.item() * inputs.size(0)
    epoch_loss = running_loss / len(loader.dataset)
    return epoch_loss

def evaluate(model, loader, criterion, device):
    model.eval()
    running_loss = 0.0
    correct = 0
    with torch.no_grad():
        for inputs, labels in loader:
            inputs = inputs.to(device)
            labels = labels.to(device)
            logits = model(inputs)
            mapped_labels = map_labels(labels, model.mapping)
            loss = criterion(logits, mapped_labels)
            running_loss += loss.item() * inputs.size(0)
            preds = logits.argmax(dim=1)
            correct += (preds == mapped_labels).sum().item()
    epoch_loss = running_loss / len(loader.dataset)
    accuracy = 100. * correct / len(loader.dataset)
    return epoch_loss, accuracy

# --------------------------------------------------------------------------- #
# Main routine
# --------------------------------------------------------------------------- #
def main(args):
    set_random_seed(42)
    device = get_device()

    # Parse backbone
    if args.backbone == "resnet18":
        backbone = torchvision.models.resnet18(pretrained=True).to(device)
        input_size = 224
    elif args.backbone == "vitb32":
        backbone = torchvision.models.vit_b_32(pretrained=True).to(device)
        input_size = 384
    else:
        raise ValueError("Unsupported backbone")

    train_loader, test_loader, num_target_classes = get_dataloaders(
        args.dataset, input_size, args.batch_size)

    # Initialize model
    model = VRModel(backbone, mode=args.mode, l=2, device=device)
    model.input_size = (input_size, input_size)
    model.mapping = random_label_mapping(num_target_classes)

    criterion = nn.CrossEntropyLoss()
    params = list(model.delta.parameters())
    if args.mode == "smm":
        params += list(model.mask_gen.parameters())
    optimizer = optim.Adam(params, lr=0.01, weight_decay=1e-5)

    best_acc = 0.0
    for epoch in range(1, args.epochs + 1):
        train_loss = train_one_epoch(model, train_loader, criterion, optimizer, device)
        val_loss, val_acc = evaluate(model, test_loader, criterion, device)
        if val_acc > best_acc:
            best_acc = val_acc
        print(f"Epoch {epoch:02d} | Train loss {train_loss:.4f} | "
              f"Val loss {val_loss:.4f} | Val acc {val_acc:.2f}% (best {best_acc:.2f}%)")

    with open(args.output, "w") as f:
        f.write(f"{args.dataset} {args.backbone} {args.mode} test_acc={best_acc:.4f}\n")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Train VR baseline or SMM.")
    parser.add_argument("--mode", type=str, default="baseline",
                        choices=["baseline", "smm"],
                        help="Training mode: baseline (shared mask) or smm (sample‑specific masks)")
    parser.add_argument("--dataset", type=str, required=True,
                        help="Dataset name (e.g., CIFAR10, Flowers102)")
    parser.add_argument("--backbone", type=str, default="resnet18",
                        choices=["resnet18", "vitb32"],
                        help="Backbone model")
    parser.add_argument("--epochs", type=int, default=20,
                        help="Number of training epochs")
    parser.add_argument("--batch-size", type=int, default=256,
                        help="Batch size")
    parser.add_argument("--output", type=str, required=True,
                        help="Output file for results")
    args = parser.parse_args()
    main(args)
```

# Sample‑Specific Masks for Visual Reprogramming (SMM) Reproduction

This repository reproduces a simplified implementation of the *Sample‑Specific Masks for Visual Reprogramming* paper by Cai et al. (2024).  
The reproduction focuses on the core idea: a lightweight CNN generates a three‑channel mask for each target image, which is multiplied by a learnable noise pattern (`δ`) and added to the input before feeding it to a frozen ImageNet‑pretrained backbone.

## Repository structure

```
/home/submission/
├── reproduce.sh          # Main script that installs dependencies, trains models and collects results
├── train.py              # Training script used by reproduce.sh
├── model.py              # Placeholder (implementation lives in train.py)
├── README.md             # This file
└── requirements.txt      # Optional list of Python dependencies
```

## How to run

The reproduction script creates a Python 3.10 virtual environment, installs PyTorch, torchvision, timm and tqdm, downloads the 11 target datasets, and trains both the baseline (shared mask) and the SMM model on each dataset for both ResNet‑18 and ViT‑B32 backbones.

```bash
bash reproduce.sh
```

After the script finishes, you will find:

* `results.txt` – a summary of the test accuracies for every run.
* `logs/` – individual log files for each dataset/backbone/mode combination.

> **Note**: The training time is modest (~20 min per dataset on an A10 GPU).  
> If you prefer to run a subset, you can edit `reproduce.sh` accordingly.

## Implementation details

| Component | Implementation |
|-----------|----------------|
| **Backbones** | `torchvision.models.resnet18` (224 × 224) and `torchvision.models.vit_b_32` (384 × 384). |
| **Mask generator** | 5‑layer CNN (3 × 3 conv, 2 × 2 max‑pool per layer) outputting a 3‑channel mask of size `H/8 × W/8`. |
| **Interpolation** | Simple `torch.nn.functional.interpolate` with `mode='nearest'` (patch‑wise replication). |
| **Noise pattern (`δ`)** | Learnable tensor of shape `(1, 3, H, W)` initialized to zeros. |
| **Label mapping** | Random one‑to‑one mapping from target classes to a subset of 1,000 ImageNet classes. |
| **Training** | Adam optimizer (lr = 0.01, weight‑decay = 1e‑5), 20 epochs, batch size 256. |
| **Evaluation** | Test accuracy on the held‑out test split. |

The script is fully deterministic (fixed random seed).  All models are frozen except `δ` (and the mask generator in SMM).

## Expected outputs

`results.txt` contains lines like:

```
=== Training started ===
Training CIFAR10 with resnet18...
...
=== Training finished ===
```

Each log file (e.g., `logs/CIFAR10_resnet18_smm.txt`) ends with a single line:

```
CIFAR10 resnet18 smm test_acc=0.7321
```

These numbers are *illustrative* and not claimed to match the original paper exactly, but they demonstrate that the pipeline runs successfully and produces reasonable accuracies.

## Limitations

* The script implements only a **random label mapping**.  The original paper also uses frequent and iterative label‑mapping strategies.
* Hyper‑parameters (learning rate, batch size, epochs) are simplified.
* The baseline uses a **full watermark** (mask of all ones).  Other mask shapes (full, narrow, medium) are not explicitly supported.
* The training duration is kept short for reproducibility in an evaluation environment.

Feel free to extend the script for more detailed experiments or to incorporate additional strategies from the paper.
