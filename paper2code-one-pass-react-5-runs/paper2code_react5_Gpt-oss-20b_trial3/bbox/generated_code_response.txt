# BBox‑Adapter Reimplementation

This repository contains a **minimal** but functional re‑implementation of the
**BBox‑Adapter** method from the paper
“Lightweight Adapting for Black‑Box Large Language Models”.
The code demonstrates the core ideas:

* A *black‑box* language model (a GPT‑2 variant) is used as a **proposal generator**.
* A small **adapter** (DeBERTa‑v3‑base) is trained to score
  candidate responses.
* Training uses a **ranking‑based Noise‑Contrastive Estimation (NCE) loss**,
  distinguishing ground‑truth (positive) from generated (negative) samples.
* An **online adaptation loop** samples new candidates each epoch and
  updates the adapter.
* Evaluation measures exact‑match accuracy on held‑out data.

> **NOTE**  
> This implementation is intentionally lightweight.  It uses only the
> publicly available GPT‑2 and DeBERTa models, does not require
> OpenAI API keys, and is fully reproducible on a single GPU.

## Quick Start

```bash
# 1. Create a virtual environment (recommended)
python -m venv venv
source venv/bin/activate

# 2. Install the dependencies
pip install -r requirements.txt

# 3. Download the datasets (cached automatically by `datasets`)
python src/dataset.py --download

# 4. Train the adapter
python src/train.py --dataset gsm8k --epochs 3 --batch_size 8

# 5. Evaluate
python src/evaluate.py --dataset gsm8k
```

The `reproduce.sh` script below runs the whole pipeline in order.

def logsumexp(tensor, dim):
    """Numerically stable log-sum-exp."""
    max_val = tensor.max(dim=dim, keepdim=True).values
    return max_val + torch.log(torch.sum(torch.exp(tensor - max_val), dim=dim, keepdim=True))

def device_of(tensor):
    return tensor.device

"""
Minimal dataset loader based on HuggingFace `datasets`.
Supports GSM8K, StrategyQA, TruthfulQA, ScienceQA.
"""

import datasets
import os
import json

from pathlib import Path

DATASETS = {
    "gsm8k": ("gsm8k", "train"),
    "strategyqa": ("strategyqa", "train"),
    "truthfulqa": ("truthfulqa", "train"),
    "scienceqa": ("scienceqa", "train"),
}

# Save a small config file so that train/eval can load the same split
CONFIG_FILE = Path("dataset_config.json")

def download_and_cache():
    """Download all datasets and store a config file."""
    cfg = {}
    for name, (repo, split) in DATASETS.items():
        ds = datasets.load_dataset(repo, split=split, cache_dir="datasets")
        # Keep only a small subset for quick runs
        ds = ds.shuffle(seed=42).select(range(2000))  # 2k examples
        cfg[name] = {
            "num_examples": len(ds),
            "fields": ds.column_names,
        }
        ds.save_to_disk(f"datasets/{name}")
    with open(CONFIG_FILE, "w") as f:
        json.dump(cfg, f, indent=2)
    print("Datasets downloaded and cached.")

def load_dataset(name, split="train"):
    """Load a dataset from disk."""
    path = Path(f"datasets/{name}")
    if not path.exists():
        raise FileNotFoundError(f"Dataset {name} not found. Run `python src/dataset.py --download` first.")
    ds = datasets.load_from_disk(path)
    return ds

def parse_args():
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--download", action="store_true")
    return parser.parse_args()

if __name__ == "__main__":
    args = parse_args()
    if args.download:
        download_and_cache()

```python
"""
Black‑box language model wrapper.
Uses GPT‑2 (medium) as the proposal generator.
"""

import torch
from transformers import GPT2LMHeadModel, GPT2TokenizerFast

class BlackBoxLLM:
    def __init__(self, model_name="gpt2-medium", device="cuda"):
        self.device = device
        self.tokenizer = GPT2TokenizerFast.from_pretrained(model_name)
        self.model = GPT2LMHeadModel.from_pretrained(model_name).to(device)
        self.model.eval()

    def generate(self, prompt, k=3, max_length=64, temperature=1.0, top_k=50):
        """
        Generate `k` candidate completions for the given prompt.
        Returns a list of strings.
        """
        input_ids = self.tokenizer(prompt, return_tensors="pt").input_ids.to(self.device)
        with torch.no_grad():
            outputs = self.model.generate(
                input_ids,
                max_new_tokens=max_length,
                temperature=temperature,
                top_k=top_k,
                do_sample=True,
                num_return_sequences=k,
                pad_token_id=self.tokenizer.eos_token_id,
            )
        # decode and strip prompt
        completions = []
        for out in outputs:
            text = self.tokenizer.decode(out, skip_special_tokens=True)
            # remove the prompt part
            completions.append(text[len(prompt):].strip())
        return completions
```

```python
"""
Adapter model and ranking‑based NCE loss.
"""

import torch
import torch.nn as nn
from transformers import DebertaV2Model, DebertaV2TokenizerFast
from torch.nn.functional import logsumexp

class Adapter(nn.Module):
    def __init__(self, base_model="microsoft/deberta-v3-base"):
        super().__init__()
        self.tokenizer = DebertaV2TokenizerFast.from_pretrained(base_model)
        self.encoder = DebertaV2Model.from_pretrained(base_model)
        self.scorer = nn.Linear(self.encoder.config.hidden_size, 1)

    def forward(self, texts):
        """
        texts: list of strings (batch)
        Returns: (batch, 1) scores
        """
        enc = self.tokenizer(texts, return_tensors="pt", padding=True, truncation=True, max_length=256)
        input_ids = enc.input_ids.to(self.encoder.device)
        attention_mask = enc.attention_mask.to(self.encoder.device)
        outputs = self.encoder(input_ids=input_ids, attention_mask=attention_mask)
        # use [CLS] token representation
        cls = outputs.last_hidden_state[:, 0, :]
        scores = self.scorer(cls).squeeze(-1)  # (batch,)
        return scores

    @staticmethod
    def nce_loss(scores_pos, scores_neg):
        """
        Ranking‑based NCE loss.
        scores_pos: (batch,)
        scores_neg: (batch, n_neg)
        """
        # Concatenate positive and negatives
        all_scores = torch.cat([scores_pos.unsqueeze(1), scores_neg], dim=1)  # (batch, 1+n)
        # Log-sum-exp
        lse = logsumexp(all_scores, dim=1)  # (batch,)
        loss = - (scores_pos - lse)  # (batch,)
        return loss.mean()
```

"""
Online adaptation training loop.
"""

import os
import torch
from tqdm import tqdm
from transformers import logging
logging.set_verbosity_error()

from .dataset import load_dataset
from .blackbox_llm import BlackBoxLLM
from .adapter import Adapter

DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")

def collate_batch(batch):
    """Return prompts and answers."""
    prompts = [ex["question"] if "question" in ex else ex["prompt"] for ex in batch]
    answers = [ex["answer"] if "answer" in ex else ex["target"] for ex in batch]
    return prompts, answers

def train_one_epoch(adapter, blackbox, dataset, optimizer, k=3, max_len=64):
    adapter.train()
    total_loss = 0.0
    for prompts, answers in tqdm(dataset, desc="Training"):
        # Generate k candidates
        candidates = []
        for p in prompts:
            cand = blackbox.generate(p, k=k, max_length=max_len)
            candidates.append(cand)
        # Build positive and negative lists
        pos_texts = []  # prompt + answer
        neg_texts = []  # prompt + each negative candidate
        for p, a, cand in zip(prompts, answers, candidates):
            pos_texts.append(f"{p} {a}")
            # exclude the positive answer if it appears among candidates
            for c in cand:
                if c.strip() == a.strip():
                    continue
                neg_texts.append(f"{p} {c}")
        if not neg_texts:
            continue  # skip if no negatives
        # Convert to tensors
        with torch.no_grad():
            pos_scores = adapter(pos_texts).to(DEVICE)
            neg_scores = adapter(neg_texts).to(DEVICE)
        # Reshape negatives back to (batch, n_neg)
        n_neg = k  # approximate
        neg_scores = neg_scores.view(-1, n_neg)
        loss = Adapter.nce_loss(pos_scores, neg_scores)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        total_loss += loss.item()
    return total_loss / len(dataset)

def main():
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--dataset", default="gsm8k", choices=["gsm8k","strategyqa","truthfulqa","scienceqa"])
    parser.add_argument("--epochs", type=int, default=3)
    parser.add_argument("--batch_size", type=int, default=8)
    parser.add_argument("--k", type=int, default=3)
    parser.add_argument("--lr", type=float, default=5e-6)
    parser.add_argument("--max_len", type=int, default=64)
    parser.add_argument("--ckpt_dir", default="checkpoints")
    args = parser.parse_args()

    os.makedirs(args.ckpt_dir, exist_ok=True)

    # Load dataset
    ds = load_dataset(args.dataset, split="train")
    # Create DataLoader
    from torch.utils.data import DataLoader
    dataloader = DataLoader(ds, batch_size=args.batch_size, shuffle=True, collate_fn=collate_batch)

    # Initialize models
    blackbox = BlackBoxLLM(device=DEVICE)
    adapter = Adapter().to(DEVICE)
    optimizer = torch.optim.AdamW(adapter.parameters(), lr=args.lr)

    for epoch in range(1, args.epochs+1):
        loss = train_one_epoch(adapter, blackbox, dataloader, optimizer,
                               k=args.k, max_len=args.max_len)
        print(f"Epoch {epoch}/{args.epochs} | Loss: {loss:.4f}")
        # Save checkpoint
        ckpt_path = os.path.join(args.ckpt_dir, f"adapter_epoch{epoch}.pt")
        torch.save(adapter.state_dict(), ckpt_path)

    print("Training finished.")

if __name__ == "__main__":
    main()

"""
Evaluate the adapter on a held‑out set.
"""

import os
import torch
from tqdm import tqdm
from transformers import logging
logging.set_verbosity_error()

from .dataset import load_dataset
from .blackbox_llm import BlackBoxLLM
from .adapter import Adapter

DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")

def collate_batch(batch):
    prompts = [ex["question"] if "question" in ex else ex["prompt"] for ex in batch]
    answers = [ex["answer"] if "answer" in ex else ex["target"] for ex in batch]
    return prompts, answers

def evaluate(adapter, blackbox, dataset, k=3, max_len=64):
    adapter.eval()
    correct = 0
    total = 0
    with torch.no_grad():
        for prompts, answers in tqdm(dataset, desc="Evaluating"):
            candidates = []
            for p in prompts:
                cand = blackbox.generate(p, k=k, max_length=max_len)
                candidates.append(cand)
            # Score each candidate
            scores = []
            for p, cand in zip(prompts, candidates):
                texts = [f"{p} {c}" for c in cand]
                s = adapter(texts).cpu().numpy()
                scores.append(s)
            # Pick the best candidate
            for i, s in enumerate(scores):
                pred_idx = s.argmax()
                pred = candidates[i][pred_idx]
                if pred.strip() == answers[i].strip():
                    correct += 1
                total += 1
    acc = correct / total if total > 0 else 0.0
    print(f"Accuracy: {acc*100:.2f}%")
    return acc

def main():
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--dataset", default="gsm8k", choices=["gsm8k","strategyqa","truthfulqa","scienceqa"])
    parser.add_argument("--ckpt_dir", default="checkpoints")
    parser.add_argument("--k", type=int, default=3)
    parser.add_argument("--max_len", type=int, default=64)
    args = parser.parse_args()

    # Load adapter checkpoint
    ckpt_path = os.path.join(args.ckpt_dir, "adapter_epoch3.pt")
    if not os.path.exists(ckpt_path):
        raise FileNotFoundError(f"Checkpoint {ckpt_path} not found. Run training first.")
    adapter = Adapter().to(DEVICE)
    adapter.load_state_dict(torch.load(ckpt_path, map_location=DEVICE))

    # Load dataset
    ds = load_dataset(args.dataset, split="validation")
    from torch.utils.data import DataLoader
    dataloader = DataLoader(ds, batch_size=8, shuffle=False, collate_fn=collate_batch)

    # Black‑box generator
    blackbox = BlackBoxLLM(device=DEVICE)

    evaluate(adapter, blackbox, dataloader, k=args.k, max_len=args.max_len)

if __name__ == "__main__":
    main()
```
