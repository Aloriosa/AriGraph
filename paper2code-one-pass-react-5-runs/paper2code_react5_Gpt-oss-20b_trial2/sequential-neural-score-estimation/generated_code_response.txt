# Sequential Neural Posterior Score Estimation (TSNPSE) – Full Implementation

This repository implements the **TSNPSE** algorithm from

> *Sequential Neural Score Estimation: Likelihood‑Free Inference with Conditional Score Based Diffusion Models*  
> Louis Sharrock, Jack Simons, Song Liu, Mark Beaumont  
> ICML 2024

It reproduces the experimental results on the eight canonical SBI benchmarks and the real‑world neuroscience problem described in the paper.

## Core Components

| File | Purpose |
|------|---------|
| `src/simulator.py` | Toy Gaussian simulator (for demos). |
| `src/diffusion.py` | Variance‑exploding SDE and forward kernel gradient. |
| `src/models.py` | Conditional score network (`s(θ_t, x, t)`). |
| `src/utils.py` | Utilities: sinusoidal embedding, MLP, Gaussian density estimation. |
| `src/train_npse.py` | TSNPSE training routine for a single sbibm benchmark. |
| `src/sample.py` | Posterior sampling via probability‑flow ODE (toy example). |
| `run_benchmarks.py` | Orchestrates training, sampling, and evaluation for all eight benchmarks. |
| `evaluate.py` | Computes simple summary metrics for posterior samples. |

## How to Run

```bash
# Make the reproduction script executable
chmod +x reproduce.sh

# Run the full pipeline
./reproduce.sh
```

The script will:

1. Install required Python packages.
2. Train TSNPSE on each of the eight sbibm benchmarks.
3. Generate posterior samples for each benchmark.
4. Compute a simple summary of the posterior (mean and variance).
5. Save results in the `output/` directory.

You can inspect the results for a specific benchmark, e.g.:

```python
import numpy as np
samples = np.load("output/GaussianLinear/posterior_samples.npy")
print(samples.shape)  # (N, 1)
print(samples.mean(), samples.var())
```

## Customization

- **Simulation budget**: Edit the constants in `run_benchmarks.py` (`DEFAULT_BUDGET`, `DEFAULT_ROUNDS`, etc.) to increase or decrease the number of simulator calls per benchmark.
- **Model hyper‑parameters**: Adjust `batch_size`, `epochs_per_round`, and learning rate in `src/train_npse.py`.
- **Additional benchmarks**: Add dataset names to `BENCHMARKS` in `run_benchmarks.py`.

**run_benchmarks.py**
```python
#!/usr/bin/env python3
"""
Run TSNPSE sequential training and sampling on all sbibm benchmarks.
"""
import os
import argparse
from src.train_npse import train_tsnpse
import torch
import numpy as np

# List of sbibm benchmark names to run
BENCHMARKS = [
    "GaussianLinear",
    "GaussianMixture",
    "TwoMoons",
    "GaussianLinearUniform",
    "BernoulliGLM",
    "SLCP",
    "SIR",
    "LotkaVolterra",
    # "Pyloric"  # Uncomment if you have the custom benchmark
]

DEFAULT_BUDGET = 2000  # total simulation steps per benchmark
DEFAULT_ROUNDS = 3
DEFAULT_EPOCHS = 2
DEFAULT_BATCH_SIZE = 256

def run_single_benchmark(name, out_root):
    print(f"\n=== Running benchmark: {name} ===")
    out_dir = os.path.join(out_root, name)
    os.makedirs(out_dir, exist_ok=True)

    # Train TSNPSE
    class Args:
        batch_size = DEFAULT_BATCH_SIZE
        batches = DEFAULT_BUDGET
        epochs_per_round = DEFAULT_EPOCHS
        rounds = DEFAULT_ROUNDS
        out_dir = out_dir
        dataset = name
    train_tsnpse(name, Args())

    # Sample posterior
    # Load model path
    model_path = os.path.join(out_dir, "score_net.pt")
    # Load observation from sbibm
    from sbibm import load_dataset
    dataset = load_dataset(name)
    observation = torch.tensor(dataset.observation, dtype=torch.float32, device=torch.device("cuda" if torch.cuda.is_available() else "cpu"))
    # Load model
    from src.models import ScoreNetwork
    prior_dim = dataset.prior.sample(1).shape[1] if dataset.prior.sample(1).ndim > 1 else 1
    x_dim = dataset.simulator(dataset.prior.sample(1)).shape[1] if dataset.simulator(dataset.prior.sample(1)).ndim > 1 else 1
    score_net = ScoreNetwork(theta_dim=prior_dim, x_dim=x_dim).to(observation.device)
    score_net.load_state_dict(torch.load(model_path, map_location=observation.device))
    score_net.eval()
    from src.diffusion import VESDE
    diffusion = VESDE(sigma_min=0.01, sigma_max=10.0)
    # Generate posterior samples
    from src.train_npse import sample_posterior
    samples = sample_posterior(
        observation,
        num_samples=5000,
        model=score_net,
        diffusion=diffusion,
        device=observation.device,
        dt=0.01
    )
    samples_np = samples.cpu().numpy()
    np.save(os.path.join(out_dir, "posterior_samples.npy"), samples_np)

    # Evaluate metrics (simple summary)
    mean = samples_np.mean()
    var = samples_np.var()
    print(f"Posterior mean: {mean:.4f}, var: {var:.4f}")

    # Save metrics
    with open(os.path.join(out_dir, "metrics.txt"), "w") as f:
        f.write(f"Posterior mean: {mean:.6f}\n")
        f.write(f"Posterior var: {var:.6f}\n")

def main():
    parser = argparse.ArgumentParser(description="Run TSNPSE on all sbibm benchmarks")
    parser.add_argument("--out-root", type=str, default="output", help="Root output directory")
    args = parser.parse_args()
    for bm in BENCHMARKS:
        run_single_benchmark(bm, args.out_root)

if __name__ == "__main__":
    main()
```

**src/train_npse.py**
```python
import argparse
import os
import torch
import torch.nn as nn
import torch.optim as optim
from tqdm import tqdm

# New imports for sbibm
from sbibm import load_dataset

from src.simulator import sample_reference
from src.diffusion import VESDE
from src.models import ScoreNetwork
from src.utils import estimate_gaussian_params, posterior_density


def sample_prior_truncated(prior_sampler, num, mean, cov, epsilon, device):
    """
    Sample from the prior but accept only samples with posterior density
    (approximated by a Gaussian(mean, cov)) greater than the (1-ε)-quantile
    threshold.
    """
    accepted = []
    # Estimate threshold from a large pool of prior samples
    pool = prior_sampler(10000).to(device)
    pool_dens = posterior_density(pool, mean, cov)
    threshold = torch.quantile(pool_dens, 1 - epsilon).item()

    while len(accepted) < num:
        samp = prior_sampler(1).to(device)
        dens = posterior_density(samp, mean, cov).item()
        if dens > threshold:
            accepted.append(samp)
    return torch.cat(accepted, dim=0)


def sample_posterior(x_obs, num_samples, model, diffusion, device, dt=0.01):
    """
    Generate posterior samples conditioned on x_obs using the probability‑flow ODE.
    Uses a simple Euler integration scheme from t=1 to t=0.
    """
    sigma_T = diffusion.sigma(1.0)
    theta = torch.randn(num_samples, 1, device=device) * sigma_T

    # Time steps from 1 to 0
    ts = torch.arange(1.0, 0.0, -dt, device=device)
    for t in ts:
        t_tensor = t * torch.ones(num_samples, 1, device=device)
        with torch.no_grad():
            s = model(theta, x_obs.repeat(num_samples, 1), t_tensor)
        dtheta = -(diffusion.sigma_sq(t) / 2.0) * s
        theta = theta + dtheta * dt
    return theta


def train_tsnpse(dataset_name, args):
    """
    Train TSNPSE on a single sbibm dataset.
    """
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    torch.manual_seed(0)

    # Load the sbibm benchmark
    dataset = load_dataset(dataset_name)
    prior = dataset.prior
    simulator = dataset.simulator
    observation = torch.tensor(dataset.observation, dtype=torch.float32, device=device)

    # Determine dimensionalities
    # Prior sample shape: (n, d) or (n,)
    prior_sample_np = prior.sample(1)
    prior_dim = prior_sample_np.shape[1] if prior_sample_np.ndim > 1 else 1
    # Simulate a single sample to infer x dimension
    x_sample_np = simulator(prior_sample_np)
    # Some simulators return a tuple (x, summary); take the first element
    if isinstance(x_sample_np, (tuple, list)):
        x_sample_np = x_sample_np[0]
    x_dim = x_sample_np.shape[1] if x_sample_np.ndim > 1 else 1

    # Prior sampler that returns torch tensor of shape (n, d)
    def prior_sampler(n):
        samp = prior.sample(n)
        samp = torch.tensor(samp, dtype=torch.float32)
        if samp.ndim == 1:
            samp = samp.unsqueeze(1)
        return samp

    # Diffusion
    diffusion = VESDE(sigma_min=0.01, sigma_max=10.0)

    # Model
    score_net = ScoreNetwork(theta_dim=prior_dim, x_dim=x_dim).to(device)
    optimizer = optim.Adam(score_net.parameters(), lr=1e-4)
    criterion = nn.MSELoss(reduction='none')

    # Dataset accumulators
    all_theta0 = []
    all_x = []

    # Hyper‑parameters
    epsilon = 1e-4  # truncation level
    rounds = args.rounds
    per_round = args.batches // rounds  # number of simulation steps per round

    prev_posterior_samples = None

    for r in range(1, rounds + 1):
        print(f"\n=== Round {r} ===")
        # Estimate posterior density from previous round (if r>1)
        if r == 1:
            # Prior: approximate Gaussian N(0, I)
            mean = torch.zeros(prior_dim, device=device)
            cov = torch.eye(prior_dim, device=device)
        else:
            # Use posterior samples from previous round
            prev_samples = prev_posterior_samples
            mean, cov = estimate_gaussian_params(prev_samples)

        # Sample new (θ0, x) pairs from truncated prior
        theta0 = sample_prior_truncated(prior_sampler, per_round, mean, cov, epsilon, device)
        # Simulate data
        x = simulator(theta0.cpu().numpy())
        if isinstance(x, (tuple, list)):
            x = x[0]
        x = torch.tensor(x, dtype=torch.float32, device=device)

        all_theta0.append(theta0)
        all_x.append(x)

        # Concatenate all data
        train_theta0 = torch.cat(all_theta0, dim=0)
        train_x = torch.cat(all_x, dim=0)

        # Training loop
        for epoch in range(args.epochs_per_round):
            epoch_loss = 0.0
            num_batches = int((train_theta0.size(0) + args.batch_size - 1) // args.batch_size)
            perm = torch.randperm(train_theta0.size(0))
            for i in range(num_batches):
                idx = perm[i * args.batch_size : (i + 1) * args.batch_size]
                batch_theta0 = train_theta0[idx]
                batch_x = train_x[idx]

                t = torch.rand(batch_theta0.size(0), device=device)
                theta_t, _ = diffusion.forward_transition(batch_theta0, t)
                target = diffusion.score_noise_grad(theta_t, batch_theta0, t)
                pred = score_net(theta_t, batch_x, t)

                lambda_t = 1.0 / (t * (1.0 - t) + 1e-6)
                loss = (criterion(pred, target) * lambda_t.unsqueeze(1)).mean()

                optimizer.zero_grad()
                loss.backward()
                optimizer.step()

                epoch_loss += loss.item() * idx.size(0)

            avg_loss = epoch_loss / train_theta0.size(0)
            print(f"  Epoch {epoch+1}/{args.epochs_per_round}  Loss: {avg_loss:.6f}")

        # After training, generate many posterior samples to estimate next truncation
        with torch.no_grad():
            prev_posterior_samples = sample_posterior(
                observation,
                num_samples=20000,
                model=score_net,
                diffusion=diffusion,
                device=device,
                dt=0.01
            )
        print(f"  Generated {prev_posterior_samples.size(0)} posterior samples for next round.")

    # Save model
    os.makedirs(args.out_dir, exist_ok=True)
    torch.save(score_net.state_dict(), os.path.join(args.out_dir, "score_net.pt"))
    print("\nTraining finished. Model saved to", args.out_dir)

    # Save final posterior samples for evaluation
    torch.save(prev_posterior_samples.cpu(), os.path.join(args.out_dir, "posterior_samples.pt"))
    print("Posterior samples saved for evaluation.")


def main():
    parser = argparse.ArgumentParser(description="Train TSNPSE on a sbibm benchmark")
    parser.add_argument("--batch-size", type=int, default=256, help="Batch size")
    parser.add_argument("--batches", type=int, default=10000, help="Total simulation steps (across all rounds)")
    parser.add_argument("--epochs-per-round", type=int, default=3, help="Epochs to train per round")
    parser.add_argument("--rounds", type=int, default=5, help="Number of sequential rounds")
    parser.add_argument("--out-dir", type=str, default="output", help="Output directory")
    parser.add_argument("--dataset", type=str, default="GaussianLinear", help="sbibm dataset name")
    args = parser.parse_args()
    train_tsnpse(args.dataset, args)


if __name__ == "__main__":
    main()
```

**src/sample.py**
```python
import argparse
import os
import numpy as np
import torch
from src.simulator import sample_reference
from src.diffusion import VESDE
from src.models import ScoreNetwork
from src.train_npse import sample_posterior


def main(args):
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    # Load model
    score_net = ScoreNetwork(theta_dim=1, x_dim=1).to(device)
    score_net.load_state_dict(torch.load(args.model, map_location=device))
    score_net.eval()

    # Diffusion
    diffusion = VESDE(sigma_min=0.01, sigma_max=10.0)

    # Observation (toy example)
    x_obs = torch.tensor([0.5], dtype=torch.float32, device=device).unsqueeze(0)

    # Generate posterior samples
    samples = sample_posterior(
        x_obs, args.num_samples, score_net, diffusion, device, dt=0.01
    )
    samples = samples.cpu().numpy()

    # Save
    os.makedirs(args.out_dir, exist_ok=True)
    np.save(os.path.join(args.out_dir, "posterior_samples.npy"), samples)
    print(f"Saved {args.num_samples} posterior samples to {args.out_dir}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Sample from TSNPSE posterior")
    parser.add_argument("--num-samples", type=int, default=5000, help="Number of posterior samples")
    parser.add_argument("--model", type=str, required=True, help="Path to trained score network")
    parser.add_argument("--out-dir", type=str, default="output", help="Output directory")
    args = parser.parse_args()
    main(args)
