def get_pplm_batch_iterator(
    tokenizer,

                data.append(json.loads(line.strip()))

    else:
        data = [d for d in data if d.get("prompt") is not None]

    def tokenize_data(batch):
        prompts = [item["prompt"] for item in batch]
        chosen = [item["chosen"] for item in batch]
        rejected = [item["rejected"] for item in batch]
        
        # Tokenize prompts
        prompt_encodings = tokenizer(
            prompts,
            max_length=max_prompt_length,
            padding="max_length",
            truncation=True,
            return_tensors="pt"
        )
        
        # Tokenize responses
        chosen_encodings = tokenizer(
            chosen,
            max_length=max_new_tokens,
            padding="max_length",
            truncation=True,
            return_tensors="pt"
        )
        
        rejected_encodings = tokenizer(
            rejected,
            max_length=max_new_tokens,
            padding="max_length",
            truncation=True,
            return_tensors="pt"
        )
        
        # Combine prompt and response
        pos_input_ids = torch.cat([prompt_encodings["input_ids"], chosen_encodings["input_ids"]], dim=1)
        pos_attention_mask = torch.cat([prompt_encodings["attention_mask"], chosen_encodings["attention_mask"]], dim=1)
        
        neg_input_ids = torch.cat([prompt_encodings["input_ids"], rejected_encodings["input_ids"]], dim=1)
        neg_attention_mask = torch.cat([prompt_encodings["attention_mask"], rejected_encodings["attention_mask"]], dim=1)
        
        # Create labels (for SFT loss)
        pos_labels = pos_input_ids.clone()
        pos_labels[:, :prompt_encodings["input_ids"].shape[1]] = -100  # Ignore prompt tokens in loss
        
        neg_labels = neg_input_ids.clone()
        neg_labels[:, :prompt_encodings["input_ids"].shape[1]] = -100  # Ignore prompt tokens in loss
        
        return {
            "pos_input_ids": pos_input_ids,
            "pos_attention_mask": pos_attention_mask,
            "pos_labels": pos_labels,
            "neg_input_ids": neg_input_ids,
            "neg_attention_mask": neg_attention_mask,
            "neg_labels": neg_labels
        }
    
    # Create batches
    for i in range(0, len(data), batch_size):
        batch = data[i:i+batch_size]
        if len(batch) == 0:
            continue
        batch_data = tokenize_data(batch)
        yield batch_data

        all_logits = model(
            concatenated_input_ids,

        all_logps = get_batch_logps(

        pos_logits = all_logits[:num_pos_samples]
        neg_logits = all_logits[num_pos_samples:]

                f"\n=====\nNew best for {self.config.validation_metric}: {self.best_val_metric}.\n=====\n"
            )
            self.patience = 0

            # Save checkpoint
            output_dir = os.path.join(self.run_dir, "checkpoints")
            os.makedirs(output_dir, exist_ok=True)
            self.save(output_dir, mean_eval_metrics)
        else:
            self.patience += 1
            if self.patience >= self.config.validation_patience:
                rank0_print("Ran out of patience, stopping training...")
                return -1

        return 0

    def clip_gradient(self):
        """Clip the gradient norm of the parameters of a non-FSDP policy."""
        return torch.nn.utils.clip_grad_norm_(
            self.policy.parameters(), self.config.max_grad_norm
        ).item()

def get_batch_logps(
    logits: torch.FloatTensor,
    input_ids: torch.LongTensor,
    average_log_prob: bool = False,
) -> torch.FloatTensor:
    """
    Compute the log probabilities of the given labels under the given logits.
    """
    # Shift so that tokens < n predict n
    shift_logits = logits[..., :-1, :].contiguous()
    shift_labels = input_ids[..., 1:].contiguous()
    
    # Create a mask for non-padding tokens
    mask = (shift_labels != -100)
    
    # Get log probabilities
    log_probs = torch.nn.functional.log_softmax(shift_logits, dim=-1)
    log_probs = torch.gather(log_probs, dim=-1, index=shift_labels.unsqueeze(-1)).squeeze(-1)
    
    # Apply mask
    log_probs = log_probs * mask.float()
    
    # Sum over sequence length
    if average_log_prob:
        return (log_probs.sum(dim=-1) / mask.sum(dim=-1)).clamp(min=1e-10)
    else:
        return log_probs.sum(dim=-1)

def dpo_loss(
    policy_pos_logps: torch.FloatTensor,
    policy_neg_logps: torch.FloatTensor,
    ref_pos_logps: torch.FloatTensor,
    ref_neg_logps: torch.FloatTensor,
    beta: float = 0.1,
    reference_free: bool = False,
) -> Tuple[torch.FloatTensor, torch.FloatTensor, torch.FloatTensor]:
    """
    Compute the DPO loss.
    """
    # Compute log odds
    policy_logratios = policy_pos_logps - policy_neg_logps
    ref_logratios = ref_pos_logps - ref_neg_logps
    
    if reference_free:
        ref_logratios = 0
    
    # DPO loss
    logits = policy_logratios - ref_logratios
    losses = -torch.nn.functional.logsigmoid(beta * logits)
    
    # Compute rewards
    pos_rewards = beta * policy_pos_logps - beta * ref_pos_logps
    neg_rewards = beta * policy_neg_logps - beta * ref_neg_logps
    
    return losses, pos_rewards, neg_rewards

def get_kl_div(
    kl_criterion,
    policy_pos_logits,
    policy_neg_logits,
    ref_pos_logits,
    ref_neg_logits,
) -> Tuple[torch.FloatTensor, torch.FloatTensor]:
    """
    Compute KL divergence between policy and reference model outputs.
    """
    # Compute KL divergence for positive samples
    pos_kl_div = kl_criterion(
        torch.nn.functional.log_softmax(policy_pos_logits, dim=-1),
        torch.nn.functional.log_softmax(ref_pos_logits, dim=-1)
    ).sum(dim=-1).mean(dim=-1)
    
    # Compute KL divergence for negative samples
    neg_kl_div = kl_criterion(
        torch.nn.functional.log_softmax(policy_neg_logits, dim=-1),
        torch.nn.functional.log_softmax(ref_neg_logits, dim=-1)
    ).sum(dim=-1).mean(dim=-1)
    
    return pos_kl_div, neg_kl_div
